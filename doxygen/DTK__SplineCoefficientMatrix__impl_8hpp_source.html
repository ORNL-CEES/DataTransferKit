56 template<class Basis,int DIM>
57 SplineCoefficientMatrix<Basis,DIM>::SplineCoefficientMatrix(
58  const Teuchos::RCP<const Tpetra::Map<int,SupportId> >& operator_map,
59  const Teuchos::ArrayView<const double>& source_centers,
60  const Teuchos::ArrayView<const SupportId>& source_center_gids,
61  const Teuchos::ArrayView<const double>& dist_source_centers,
62  const Teuchos::ArrayView<const SupportId>& dist_source_center_gids,
63  const SplineInterpolationPairing<DIM>& source_pairings,
64  const Basis& basis )
65 {
66  DTK_CHECK( 0 == source_centers.size() % DIM );
67  DTK_CHECK( source_centers.size() / DIM ==
68  source_center_gids.size() );
69  DTK_CHECK( 0 == dist_source_centers.size() % DIM );
70  DTK_CHECK( dist_source_centers.size() / DIM ==
71  dist_source_center_gids.size() );
72 
73  // Get the number of source centers.
74  unsigned num_source_centers = source_center_gids.size();
75 
76  // Create the P matrix.
77  int offset = DIM + 1;
78  Teuchos::RCP<Tpetra::MultiVector<double,int,SupportId> > P_vec =
79  Tpetra::createMultiVector<double,int,SupportId>( operator_map, offset );
80  int di = 0;
81  for ( unsigned i = 0; i < num_source_centers; ++i )
82  {
83  P_vec->replaceGlobalValue( source_center_gids[i], 0, 1.0 );
84  di = DIM*i;
85  for ( int d = 0; d < DIM; ++d )
86  {
87  P_vec->replaceGlobalValue(
88  source_center_gids[i], d+1, source_centers[di+d] );
89  }
90  }
91  d_P =Teuchos::rcp( new PolynomialMatrix(P_vec,operator_map,operator_map) );
92 
93  // Create the M matrix.
94  Teuchos::ArrayRCP<SupportId> children_per_parent =
95  source_pairings.childrenPerParent();
96  SupportId max_entries_per_row = *std::max_element(
97  children_per_parent.begin(), children_per_parent.end() );
98  d_M = Teuchos::rcp( new Tpetra::CrsMatrix<double,int,SupportId>(
99  operator_map, max_entries_per_row) );
100  Teuchos::Array<SupportId> M_indices( max_entries_per_row );
101  Teuchos::Array<double> values( max_entries_per_row );
102  int dj = 0;
103  Teuchos::ArrayView<const unsigned> source_neighbors;
104  double dist = 0.0;
105  int nsn = 0;
106  double radius = 0.0;
107  for ( unsigned i = 0; i < num_source_centers; ++i )
108  {
109  // Get the source points neighboring this source point.
110  di = DIM*i;
111  source_neighbors = source_pairings.childCenterIds( i );
112  nsn = source_neighbors.size();
113  radius = source_pairings.parentSupportRadius( i );
114 
115  // Add the local basis contributions.
116  for ( int j = 0; j < nsn; ++j )
117  {
118  dj = DIM*source_neighbors[j];
119  M_indices[j] =
120  dist_source_center_gids[ source_neighbors[j] ];
121 
122  dist = EuclideanDistance<DIM>::distance(
123  &source_centers[di], &dist_source_centers[dj] );
124 
125  values[j] = BP::evaluateValue( basis, radius, dist );
126  }
127  d_M->insertGlobalValues(
128  source_center_gids[i], M_indices(0,nsn), values(0,nsn) );
129  }
130  d_M->fillComplete();
131 
132  DTK_ENSURE( d_M->isFillComplete() );
133 }
