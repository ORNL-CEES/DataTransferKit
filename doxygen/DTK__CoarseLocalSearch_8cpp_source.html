54 CoarseLocalSearch::CoarseLocalSearch(
55  const EntityIterator& entity_iterator,
56  const Teuchos::RCP<EntityLocalMap>& local_map,
57  const Teuchos::ParameterList& parameters )
58 {
59  // Setup the centroid array. These will be interleaved.
60  int space_dim = 0;
61  int num_entity = entity_iterator.size();
62  if ( num_entity > 0 )
63  {
64  space_dim = entity_iterator.begin()->physicalDimension();
65  }
66  d_entity_centroids.resize( space_dim * num_entity );
67 
68  // Add the centroids.
69  EntityIterator entity_it;
70  EntityIterator begin_it = entity_iterator.begin();
71  EntityIterator end_it = entity_iterator.end();
72  int entity_local_id = 0;
73  for ( entity_it = begin_it;
74  entity_it != end_it;
75  ++entity_it )
76  {
77  local_map->centroid(
78  *entity_it,
79  d_entity_centroids(space_dim*entity_local_id,space_dim) );
80  d_entity_map.emplace( entity_local_id, *entity_it );
81  ++entity_local_id;
82  }
83 
84  // Build a static search tree.
85  int leaf_size = 20;
86  if ( parameters.isParameter("Coarse Local Search Leaf Size") )
87  {
88  leaf_size = parameters.get<int>("Coarse Local Search Leaf Size");
89  }
90  leaf_size = std::min( leaf_size, num_entity );
91  d_tree = SearchTreeFactory::createStaticTree(
92  space_dim, d_entity_centroids(), leaf_size );
93  DTK_ENSURE( Teuchos::nonnull(d_tree) );
94 }

100 void CoarseLocalSearch::search( const Teuchos::ArrayView<const double>& point,
101  const Teuchos::ParameterList& parameters,
102  Teuchos::Array<Entity>& neighbors ) const
103 {
104  // Find the leaf of nearest neighbors.
105  int num_neighbors = 100;
106  if ( parameters.isParameter("Coarse Local Search kNN") )
107  {
108  num_neighbors = parameters.get<int>("Coarse Local Search kNN");
109  }
110  num_neighbors =
111  std::min( num_neighbors, Teuchos::as<int>(d_entity_map.size()) );
112  Teuchos::Array<unsigned> local_neighbors =
113  d_tree->nnSearch( point, num_neighbors );
114 
115  // Extract the neighbors.
116  neighbors.resize( local_neighbors.size() );
117  Teuchos::Array<unsigned>::const_iterator local_it;
118  Teuchos::Array<Entity>::iterator entity_it;
119  for ( local_it = local_neighbors.begin(),
120  entity_it = neighbors.begin();
121  local_it != local_neighbors.end();
122  ++local_it, ++entity_it )
123  {
124  DTK_CHECK( d_entity_map.count(*local_it) );
125  *entity_it = d_entity_map.find(*local_it)->second;
126  }
127 }
