174 template<int DIM>
175 Teuchos::Array<unsigned> NanoflannTree<DIM>::radiusSearch(
176  const Teuchos::ArrayView<const double>& point,
177  const double radius ) const
178 {
179  DTK_REQUIRE( DIM == point.size() );
180  Teuchos::Array<std::pair<unsigned,double> > neighbor_pairs;
181  nanoflann::SearchParams params;
182  double l2_radius = radius*radius;
183  d_tree->radiusSearch(
184  point.getRawPtr(), l2_radius, neighbor_pairs, params );
185 
186  Teuchos::Array<std::pair<unsigned,double> >::const_iterator pair_it;
187  Teuchos::Array<unsigned> neighbors( neighbor_pairs.size() );
188  Teuchos::Array<unsigned>::iterator id_it;
189  for ( id_it = neighbors.begin(),
190  pair_it = neighbor_pairs.begin();
191  id_it != neighbors.end();
192  ++id_it, ++pair_it )
193  {
194  *id_it = pair_it->first;
195  }
196 
197  return neighbors;
198 }
