230 bool BoxGeometryImpl::checkPointInclusion(
231  const double tolerance,
232  const Teuchos::ArrayView<const double>& reference_point ) const
233 {
234  DTK_REQUIRE( 3 == reference_point.size() );
235 
236  double x_tol = (d_x_max - d_x_min)*tolerance;
237  double y_tol = (d_y_max - d_y_min)*tolerance;
238  double z_tol = (d_z_max - d_z_min)*tolerance;
239 
240  if ( reference_point[0] >= d_x_min - x_tol &&
241  reference_point[1] >= d_y_min - y_tol &&
242  reference_point[2] >= d_z_min - z_tol &&
243  reference_point[0] <= d_x_max + x_tol &&
244  reference_point[1] <= d_y_max + y_tol &&
245  reference_point[2] <= d_z_max + z_tol )
246  {
247  return true;
248  }
249 
250  return false;
251 }
