59 // Constructor.
60 template<class Scalar, class FieldType>
61 STKMeshField<Scalar,FieldType>::STKMeshField(
62  const Teuchos::RCP<stk::mesh::BulkData>& bulk_data,
63  const Teuchos::Ptr<FieldType>& field,
64  const int field_dim )
65  : d_bulk_data( bulk_data )
66  , d_field( field )
67  , d_field_dim( field_dim )
68 {
69  // Get the field restriction.
70  const stk::mesh::FieldRestrictionVector field_restrictions =
71  d_field->restrictions();
72 
73  // Build a composite selector.
74  stk::mesh::Selector field_selector;
75  for ( stk::mesh::FieldRestriction r : field_restrictions )
76  {
77  field_selector = field_selector | r.selector();
78  }
79 
80  // Get the buckets for the field entity rank.
81  const stk::mesh::BucketVector& field_buckets =
82  field_selector.get_buckets( d_field->entity_rank() );
83 
84  // Get the local entities over which the field is defined.
85  stk::mesh::get_selected_entities(
86  field_selector, field_buckets, d_field_entities );
87 
88  // Map the field entity ids.
89  int num_entities = d_field_entities.size();
90  for ( int n = 0; n < num_entities; ++n )
91  {
92  d_id_map.emplace( d_bulk_data->identifier(d_field_entities[n]), n );
93  }
94 
95  // Get the locally owned field entity ids.
96  for ( int n = 0; n < num_entities; ++n )
97  {
98  if ( d_bulk_data->parallel_owner_rank(d_field_entities[n]) ==
99  d_bulk_data->parallel_rank() )
100  {
101  d_support_ids.push_back(
102  d_bulk_data->identifier(d_field_entities[n]) );
103  }
104  }
105 }

127 template<class Scalar, class FieldType>
128 double STKMeshField<Scalar,FieldType>::readFieldData(
129  const SupportId support_id, const int dimension ) const
130 {
131  DTK_REQUIRE( d_id_map.count(support_id) );
