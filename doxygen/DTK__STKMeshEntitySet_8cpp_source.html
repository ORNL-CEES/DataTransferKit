111 //---------------------------------------------------------------------------//
112 // Given an entity, get the entities of the given type that are adjacent to
113 // it.
114 void STKMeshEntitySet::getAdjacentEntities(
115  const Entity& entity,
116  const int adjacent_dimension,
117  Teuchos::Array<Entity>& adjacent_entities ) const
118 {
119  const stk::mesh::Entity& stk_entity = STKMeshHelpers::extractEntity(entity);
120  stk::mesh::EntityRank rank =
121  STKMeshHelpers::getRankFromTopologicalDimension(
122  adjacent_dimension, physicalDimension() );
123  const stk::mesh::Entity* begin =
124  d_bulk_data->begin( stk_entity, rank );
125  const stk::mesh::Entity* end = d_bulk_data->end( stk_entity, rank );
126  Teuchos::Array<stk::mesh::Entity> stk_adjacencies( begin, end );
127  adjacent_entities.resize( stk_adjacencies.size() );
128  Teuchos::Array<Entity>::iterator entity_it;
129  Teuchos::Array<stk::mesh::Entity>::iterator stk_it;
130  for ( entity_it = adjacent_entities.begin(),
131  stk_it = stk_adjacencies.begin();
132  entity_it != adjacent_entities.end();
133  ++entity_it, ++stk_it )
134  {
135  *entity_it = STKMeshEntity( *stk_it, d_bulk_data.ptr() );
136  }
137 }
