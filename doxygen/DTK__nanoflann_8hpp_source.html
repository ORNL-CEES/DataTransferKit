101  inline void addPoint(DistanceType dist, IndexType index)
102  {
103  CountType i;
104  for (i=count; i>0; --i) {
105 #ifdef NANOFLANN_FIRST_MATCH // If defined and two poins have the same distance, the one with the lowest-index will be returned first.
106  if ( (dists[i-1]>dist) || ((dist==dists[i-1])&&(indices[i-1]>index)) ) {
107 #else
108  if (dists[i-1]>dist) {
109 #endif
110  if (i<capacity) {
111  dists[i] = dists[i-1];
112  indices[i] = indices[i-1];
113  }
114  }
115  else break;
116  }
117  if (i<capacity) {
118  dists[i] = dist;
119  indices[i] = index;
120  }
121  if (count<capacity) count++;
122  }
123 
124  inline DistanceType worstDist() const
125  {
126  return dists[capacity-1];
127  }

264  inline DistanceType operator()(const T* a, const size_t b_idx, size_t size, DistanceType worst_dist = -1) const
265  {
266  DistanceType result = DistanceType();
267  const T* last = a + size;
268  const T* lastgroup = last - 3;
269  size_t d = 0;
270 
271  /* Process 4 items with each loop for efficiency. */
272  while (a < lastgroup) {
273  const DistanceType diff0 = nanoflann::abs(a[0] - data_source.kdtree_get_pt(b_idx,d++));
274  const DistanceType diff1 = nanoflann::abs(a[1] - data_source.kdtree_get_pt(b_idx,d++));
275  const DistanceType diff2 = nanoflann::abs(a[2] - data_source.kdtree_get_pt(b_idx,d++));
276  const DistanceType diff3 = nanoflann::abs(a[3] - data_source.kdtree_get_pt(b_idx,d++));
277  result += diff0 + diff1 + diff2 + diff3;
278  a += 4;
279  if ((worst_dist>0)&&(result>worst_dist)) {
280  return result;
281  }
282  }
283  /* Process last 0-3 components. Not needed for standard vector lengths. */
284  while (a < last) {
285  result += nanoflann::abs( *a++ - data_source.kdtree_get_pt(b_idx,d++) );
286  }
287  return result;
288  }

466  allocations be in multiples of the machine wordsize. */

504  {

517  void* malloc(const size_t req_size)
518  {
519  /* Round size up to a multiple of wordsize. The following expression
520  only works for WORDSIZE that is a power of 2, by masking last bits of
521  incremented size to zero.
522  */
523  const size_t size = (req_size + (WORDSIZE - 1)) & ~(WORDSIZE - 1);
524 
525  /* Check whether a new block must be allocated. Note that the first word
526  of a block is reserved for a pointer to the previous block.
527  */
528  if (size > remaining) {
529 
530  wastedMemory += remaining;
531 
532  /* Allocate new storage. */
533  const size_t blocksize = (size + sizeof(void*) + (WORDSIZE-1) > BLOCKSIZE) ?
534  size + sizeof(void*) + (WORDSIZE-1) : BLOCKSIZE;
535 
536  // use the standard C malloc to allocate memory
537  void* m = ::malloc(blocksize);
538  if (!m) {
539  fprintf(stderr,"Failed to allocate memory.\n");
540  return NULL;
541  }
542 
543  /* Fill first word of new block with pointer to previous block. */
544  ((void**) m)[0] = base;
545  base = m;
546 
547  size_t shift = 0;
548  //int size_t = (WORDSIZE - ( (((size_t)m) + sizeof(void*)) & (WORDSIZE-1))) & (WORDSIZE-1);
549 
550  remaining = blocksize - sizeof(void*) - shift;
551  loc = ((char*)m + sizeof(void*) + shift);
552  }
553  void* rloc = loc;
554  loc = (char*)loc + size;
555  remaining -= size;
556 
557  usedMemory += size;
558 
559  return rloc;
560  }

1271  distance_vector_t& dists, const float epsError) const

1275  //count_leaf += (node->lr.right-node->lr.left); // Removed since was neither used nor returned to the user.

1286

1291  DistanceType diff2 = val - node->sub.divhigh;

1294  NodePtr otherChild;
1295  DistanceType cut_dist;
1296  if ((diff1+diff2)<0) {
1297  bestChild = node->child1;
1298  otherChild = node->child2;
1299  cut_dist = distance.accum_dist(val, node->sub.divhigh, idx);
1300  }
1301  else {
1302  bestChild = node->child2;
1303  otherChild = node->child1;
1304  cut_dist = distance.accum_dist( val, node->sub.divlow, idx);
1305  }
1306 
1307  /* Call recursively to search next level down. */
1308  searchLevel(result_set, vec, bestChild, mindistsq, dists, epsError);
1309 
1310  DistanceType dst = dists[idx];
1311  mindistsq = mindistsq + cut_dist - dst;

1313  if (mindistsq*epsError<=result_set.worstDist()) {
