216 bool CylinderGeometryImpl::checkPointInclusion(
217  const double tolerance,
218  const Teuchos::ArrayView<const double>& reference_point ) const
219 {
220  DTK_REQUIRE( reference_point.size() == 3 );
221 
222  double x_dist = d_centroid_x - reference_point[0];
223  double y_dist = d_centroid_y - reference_point[1];
224  double r = std::sqrt( x_dist*x_dist + y_dist*y_dist );
225  double rad_tol = d_radius*tolerance;
226  double half_length_tol = d_length / 2 + d_length*tolerance;
227 
228  return ( (r <= d_radius + rad_tol) &&
229  (reference_point[2] >= d_centroid_z - half_length_tol) &&
230  (reference_point[2] <= d_centroid_z + half_length_tol) );
231 }
