74 template<class Basis,int DIM>
75 SplineInterpolationOperator<Basis,DIM>::SplineInterpolationOperator(
76  const Teuchos::RCP<const TpetraMap>& domain_map,
77  const Teuchos::RCP<const TpetraMap>& range_map,
78  const Teuchos::ParameterList& parameters )
79  : Base( domain_map, range_map )
80  , d_use_knn( false )
81  , d_knn( 0 )
82  , d_radius( 0.0 )
83  , d_domain_entity_dim( 0 )
84  , d_range_entity_dim( 0 )
85 {
86  // Determine if we are doing kNN search or radius search.
87  if( parameters.isParameter("Type of Search") )
88  {
89  if ( "Radius" == parameters.get<std::string>("Type of Search") )
90  {
91  d_use_knn = false;
92  }
93  else if ( "Nearest Neighbor" == parameters.get<std::string>("Type of Search") )
94  {
95  d_use_knn = true;
96  }
97  else
98  {
99  // Otherwise we got an invalid search type.
100  DTK_INSIST( false );
101  }
102  }
103 
104  // If we are doing kNN support get the number of neighbors.
105  if( d_use_knn )
106  {
107  DTK_REQUIRE( parameters.isParameter("Num Neighbors") );
108  d_knn = parameters.get<int>("Num Neighbors");
109  }
110 
111  // Otherwise we are doing the radius search so get the basis radius.
112  else
113  {
114  DTK_REQUIRE( parameters.isParameter("RBF Radius") );
115  d_radius = parameters.get<double>("RBF Radius");
116  }
117 
118  // Get the topological dimension of the domain and range entities. This
119  // map will use their centroids for the point cloud.
120  if ( parameters.isParameter("Domain Entity Dimension") )
121  {
122  d_domain_entity_dim = parameters.get<int>("Domain Entity Dimension");
123  }
124  if ( parameters.isParameter("Range Entity Dimension") )
125  {
126  d_range_entity_dim = parameters.get<int>("Range Entity Dimension");
127  }
128 
129  // Get the stratimikos parameters if they exist.
130  if ( parameters.isSublist("Stratimikos") )
131  {
132  d_stratimikos_list = Teuchos::rcp(
133  new Teuchos::ParameterList(parameters.sublist("Stratimikos")) );
134  }
135 }

139 template<class Basis,int DIM>
140 void SplineInterpolationOperator<Basis,DIM>::setupImpl(
141  const Teuchos::RCP<FunctionSpace>& domain_space,
142  const Teuchos::RCP<FunctionSpace>& range_space )
143 {
144  DTK_REQUIRE( Teuchos::nonnull(domain_space) );
145  DTK_REQUIRE( Teuchos::nonnull(range_space) );
146 
147  // Extract the Support maps.
148  const Teuchos::RCP<const typename Base::TpetraMap> domain_map
149  = this->getDomainMap();
150  const Teuchos::RCP<const typename Base::TpetraMap> range_map
151  = this->getRangeMap();
152 
153  // Prolongation operator.
154  Teuchos::RCP<const Root> S;
155 
156  // Coefficient matrix polynomial component.
157  Teuchos::RCP<const Root> P;
158 
159  // Coefficient matrix basis component.
160  Teuchos::RCP<const Root> M;
161 
162  // Evaluation matrix polynomial component.
163  Teuchos::RCP<const Root> Q;
164 
165  // Evaluation matrix basis component.
166  Teuchos::RCP<const Root> N;
167 
168  // Build the concrete operators.
169  buildConcreteOperators( domain_space, range_space, S, P, M, Q, N );
170 
171  // Create an abstract wrapper for S.
172  Teuchos::RCP<const Thyra::VectorSpaceBase<Scalar> > thyra_range_vector_space_S =
173  Thyra::createVectorSpace<Scalar>( S->getRangeMap() );
174  Teuchos::RCP<const Thyra::VectorSpaceBase<Scalar> > thyra_domain_vector_space_S =
175  Thyra::createVectorSpace<Scalar>( S->getDomainMap() );
176  Teuchos::RCP<const Thyra::TpetraLinearOp<Scalar,LO,GO> > thyra_S =
177  Teuchos::rcp( new Thyra::TpetraLinearOp<Scalar,LO,GO>() );
178  Teuchos::rcp_const_cast<Thyra::TpetraLinearOp<Scalar,LO,GO> >(
179  thyra_S)->constInitialize(
180  thyra_range_vector_space_S, thyra_domain_vector_space_S, S );
181 
182  // Create an abstract wrapper for P.
183  Teuchos::RCP<const Thyra::VectorSpaceBase<Scalar> > thyra_range_vector_space_P =
184  Thyra::createVectorSpace<Scalar>( P->getRangeMap() );
185  Teuchos::RCP<const Thyra::VectorSpaceBase<Scalar> > thyra_domain_vector_space_P =
186  Thyra::createVectorSpace<Scalar>( P->getDomainMap() );
187  Teuchos::RCP<const Thyra::TpetraLinearOp<Scalar,LO,GO> > thyra_P =
188  Teuchos::rcp( new Thyra::TpetraLinearOp<Scalar,LO,GO>() );
189  Teuchos::rcp_const_cast<Thyra::TpetraLinearOp<Scalar,LO,GO> >(
190  thyra_P)->constInitialize(
191  thyra_range_vector_space_P, thyra_domain_vector_space_P, P );
192 
193  // Create an abstract wrapper for M.
194  Teuchos::RCP<const Thyra::VectorSpaceBase<Scalar> > thyra_range_vector_space_M =
195  Thyra::createVectorSpace<Scalar>( M->getRangeMap() );
196  Teuchos::RCP<const Thyra::VectorSpaceBase<Scalar> > thyra_domain_vector_space_M =
197  Thyra::createVectorSpace<Scalar>( M->getDomainMap() );
198  Teuchos::RCP<const Thyra::TpetraLinearOp<Scalar,LO,GO> > thyra_M =
199  Teuchos::rcp( new Thyra::TpetraLinearOp<Scalar,LO,GO>() );
200  Teuchos::rcp_const_cast<Thyra::TpetraLinearOp<Scalar,LO,GO> >(
201  thyra_M)->constInitialize(
202  thyra_range_vector_space_M, thyra_domain_vector_space_M, M );
203 
204  // Create an abstract wrapper for Q.
205  Teuchos::RCP<const Thyra::VectorSpaceBase<Scalar> > thyra_range_vector_space_Q =
206  Thyra::createVectorSpace<Scalar>( Q->getRangeMap() );
207  Teuchos::RCP<const Thyra::VectorSpaceBase<Scalar> > thyra_domain_vector_space_Q =
208  Thyra::createVectorSpace<Scalar>( Q->getDomainMap() );
209  Teuchos::RCP<const Thyra::TpetraLinearOp<Scalar,LO,GO> > thyra_Q =
210  Teuchos::rcp( new Thyra::TpetraLinearOp<Scalar,LO,GO>() );
211  Teuchos::rcp_const_cast<Thyra::TpetraLinearOp<Scalar,LO,GO> >(
212  thyra_Q)->constInitialize(
213  thyra_range_vector_space_Q, thyra_domain_vector_space_Q, Q );
214 
215  // Create an abstract wrapper for N.
216  Teuchos::RCP<const Thyra::VectorSpaceBase<Scalar> > thyra_range_vector_space_N =
217  Thyra::createVectorSpace<Scalar>( N->getRangeMap() );
218  Teuchos::RCP<const Thyra::VectorSpaceBase<Scalar> > thyra_domain_vector_space_N =
219  Thyra::createVectorSpace<Scalar>( N->getDomainMap() );
220  Teuchos::RCP<const Thyra::TpetraLinearOp<Scalar,LO,GO> > thyra_N =
221  Teuchos::rcp( new Thyra::TpetraLinearOp<Scalar,LO,GO>() );
222  Teuchos::rcp_const_cast<Thyra::TpetraLinearOp<Scalar,LO,GO> >(
223  thyra_N)->constInitialize(
224  thyra_range_vector_space_N, thyra_domain_vector_space_N, N );
225 
226  // COUPLING MATRIX ASSEMBLY: A = (Q + N)*[(P + M + P^T)^-1]*S
227  // Create a transpose of P.
228  Teuchos::RCP<const Thyra::LinearOpBase<Scalar> > thyra_P_T =
229  Thyra::transpose<Scalar>( thyra_P );
230 
231  // Create a composite operator C = (P + M + P^T)
232  Teuchos::RCP<const Thyra::LinearOpBase<Scalar> > thyra_PpM =
233  Thyra::add<Scalar>( thyra_P, thyra_M );
234  Teuchos::RCP<const Thyra::LinearOpBase<Scalar> > thyra_C =
235  Thyra::add<Scalar>( thyra_PpM, thyra_P_T );
236 
237  // If we didnt get stratimikos parameters from the input list, create some
238  // here.
239  if ( Teuchos::is_null(d_stratimikos_list) )
240  {
241  d_stratimikos_list = Teuchos::parameterList("Stratimikos");
242 
243  d_stratimikos_list->set( "Linear Solver Type", "Belos" );
244  d_stratimikos_list->set( "Preconditioner Type", "None" );
245 
246  auto& linear_solver_types_list =
247  d_stratimikos_list->sublist("Linear Solver Types");
248  auto& belos_list = linear_solver_types_list.sublist( "Belos" );
249  belos_list.set( "Solver Type", "Pseudo Block GMRES" );
250  auto& solver_types_list = belos_list.sublist( "Solver Types" );
251  auto& gmres_list = solver_types_list.sublist("Pseudo Block GMRES");
252  gmres_list.set("Convergence Tolerance", 1.0e-10 );
253  gmres_list.set("Verbosity",
254  Belos::Errors + Belos::Warnings +
255  Belos::TimingDetails + Belos::FinalSummary +
256  Belos::StatusTestDetails );
257  gmres_list.set("Output Frequency", 1 );
258  }
259 
260  // Create the inverse of the composite operator C.
261  Stratimikos::DefaultLinearSolverBuilder builder;
262  builder.setParameterList( d_stratimikos_list );
263  Teuchos::RCP<Thyra::LinearOpWithSolveFactoryBase<Scalar> > factory =
264  Thyra::createLinearSolveStrategy( builder );
265  Teuchos::RCP<const Thyra::LinearOpBase<Scalar> > thyra_C_inv =
266  Thyra::inverse<Scalar>( *factory, thyra_C );
267 
268  // Create the composite operator B = (Q + N);
269  Teuchos::RCP<const Thyra::LinearOpBase<Scalar> > thyra_B =
270  Thyra::add<Scalar>( thyra_Q, thyra_N );
271 
272  // Create the coupling matrix A = (B * C^-1 * S).
273  d_coupling_matrix =
274  Thyra::multiply<Scalar>( thyra_B, thyra_C_inv, thyra_S );
275  DTK_ENSURE( Teuchos::nonnull(d_coupling_matrix) );
276 }

309 template<class Basis,int DIM>
310 void SplineInterpolationOperator<Basis,DIM>::buildConcreteOperators(
311  const Teuchos::RCP<FunctionSpace>& domain_space,
312  const Teuchos::RCP<FunctionSpace>& range_space,
313  Teuchos::RCP<const Root>& S,
314  Teuchos::RCP<const Root>& P,
315  Teuchos::RCP<const Root>& M,
316  Teuchos::RCP<const Root>& Q,
317  Teuchos::RCP<const Root>& N ) const
318 {
319  // Extract the Support maps.
320  const Teuchos::RCP<const typename Base::TpetraMap> domain_map
321  = this->getDomainMap();
322  const Teuchos::RCP<const typename Base::TpetraMap> range_map
323  = this->getRangeMap();
324 
325  // Determine if we have range and domain data on this process.
326  bool nonnull_domain = Teuchos::nonnull( domain_space->entitySet() );
327  bool nonnull_range = Teuchos::nonnull( range_space->entitySet() );
328 
329  // Get the parallel communicator.
330  Teuchos::RCP<const Teuchos::Comm<int> > comm = domain_map->getComm();
331 
332  // Extract the source nodes and their ids.
333  Teuchos::ArrayRCP<double> source_centers;
334  Teuchos::ArrayRCP<GO> source_support_ids;
335  getNodeCoordsAndIds( domain_space, d_domain_entity_dim,
336  source_centers, source_support_ids );
337 
338  // Extract the target nodes and their ids.
339  Teuchos::ArrayRCP<double> target_centers;
340  Teuchos::ArrayRCP<GO> target_support_ids;
341  getNodeCoordsAndIds( range_space, d_range_entity_dim,
342  target_centers, target_support_ids );
343 
344  // Calculate an approximate neighborhood distance for the local source
345  // centers. If using kNN, compute an approximation. If doing a radial
346  // search, use the radius.
347  double source_proximity = 0.0;
348  if ( d_use_knn )
349  {
350  // Get the local bounding box.
351  Teuchos::Tuple<double,6> local_box;
352  domain_space->entitySet()->localBoundingBox( local_box );
353 
354  // Calculate the largest span of the cardinal directions.
355  source_proximity = local_box[3] - local_box[0];
356  for ( int d = 1; d < DIM; ++d )
357  {
358  source_proximity = std::max( source_proximity,
359  local_box[d+3] - local_box[d] );
360  }
361 
362  // Take the proximity to be 10% of the largest distance.
363  source_proximity *= 0.1;
364  }
365  else
366  {
367  source_proximity = d_radius;
368  }
369 
370  // Gather the source centers that are in the proximity of the source
371  // centers on this proc.
372  Teuchos::Array<double> dist_sources;
373  CenterDistributor<DIM> source_distributor(
374  comm, source_centers(), source_centers(), source_proximity, dist_sources );
375 
376  // Distribute the global source ids.
377  Teuchos::Array<GO> dist_source_support_ids(
378  source_distributor.getNumImports() );
379  Teuchos::ArrayView<const GO> source_support_ids_view = source_support_ids();
380  Teuchos::ArrayView<GO> dist_gids_view = dist_source_support_ids();
381  source_distributor.distribute( source_support_ids_view, dist_gids_view );
382 
383  // Build the source/source pairings.
384  SplineInterpolationPairing<DIM> source_pairings(
385  dist_sources(), source_centers(), d_use_knn, d_knn, d_radius );
386 
387  // Build the basis.
388  Teuchos::RCP<Basis> basis = BP::create();
389 
390  // PROLONGATION OPERATOR.
391  GO offset = comm->getRank() ? 0 : DIM + 1;
392  S = Teuchos::rcp( new SplineProlongationOperator(offset,domain_map) );
393 
394  // Get the operator map.
395  Teuchos::RCP<const Tpetra::Map<int,GO> > prolongated_map = S->getRangeMap();
396 
397  // COEFFICIENT OPERATORS.
398  // Build the coefficient operators.
399  SplineCoefficientMatrix<Basis,DIM> C(
400  prolongated_map,
401  source_centers(), source_support_ids(),
402  dist_sources(), dist_source_support_ids(),
403  source_pairings, *basis );
404  P = C.getP();
405  M = C.getM();
406 
407  // Cleanup.
408  dist_source_support_ids.clear();
409 
410  // EVALUATION OPERATORS.
411  // Calculate an approximate neighborhood distance for the local target
412  // centers. If using kNN, compute an approximation. If doing a radial
413  // search, use the radius.
414  double target_proximity = 0.0;
415  if ( d_use_knn )
416  {
417  // Get the local bounding box.
418  Teuchos::Tuple<double,6> local_box;
419  range_space->entitySet()->localBoundingBox( local_box );
420 
421  // Calculate the largest span of the cardinal directions.
422  target_proximity = local_box[3] - local_box[0];
423  for ( int d = 1; d < DIM; ++d )
424  {
425  target_proximity = std::max( target_proximity,
426  local_box[d+3] - local_box[d] );
427  }
428 
429  // Take the proximity to be 10% of the largest distance.
430  target_proximity *= 0.1;
431  }
432  else
433  {
434  target_proximity = d_radius;
435  }
436 
437  // Gather the source centers that are in the proximity of the target
438  // centers on this proc.
439  CenterDistributor<DIM> target_distributor(
440  comm, source_centers(), target_centers(), target_proximity, dist_sources );
441 
442  // Distribute the global source ids.
443  dist_source_support_ids.resize(
444  target_distributor.getNumImports() );
445  source_support_ids_view = source_support_ids();
446  dist_gids_view = dist_source_support_ids();
447  target_distributor.distribute( source_support_ids_view, dist_gids_view );
448 
449  // Build the source/target pairings.
450  SplineInterpolationPairing<DIM> target_pairings(
451  dist_sources(), target_centers(), d_use_knn, d_knn, d_radius );
452 
453  // Build the transformation operators.
454  SplineEvaluationMatrix<Basis,DIM> B(
455  prolongated_map, range_map,
456  target_centers(), target_support_ids(),
457  dist_sources(), dist_source_support_ids(),
458  target_pairings, *basis );
459  N = B.getN();
460  Q = B.getQ();
461 
462  DTK_ENSURE( Teuchos::nonnull(S) );
463  DTK_ENSURE( Teuchos::nonnull(P) );
464  DTK_ENSURE( Teuchos::nonnull(M) );
465  DTK_ENSURE( Teuchos::nonnull(Q) );
466  DTK_ENSURE( Teuchos::nonnull(N) );
467 }

469 //---------------------------------------------------------------------------//
470 // Extract node coordinates and ids from an iterator.
471 template<class Basis,int DIM>
472 void SplineInterpolationOperator<Basis,DIM>::getNodeCoordsAndIds(
473  const Teuchos::RCP<FunctionSpace>& space,
474  const int entity_dim,
475  Teuchos::ArrayRCP<double>& centers,
476  Teuchos::ArrayRCP<GO>& support_ids ) const
477 {
478  // Get an iterator over the local nodes.
479  EntityIterator iterator;
480  if ( Teuchos::nonnull(space->entitySet()) )
481  {
482  LocalEntityPredicate local_predicate(
483  space->entitySet()->communicator()->getRank() );
484  PredicateFunction predicate =
485  PredicateComposition::And(
486  space->selectFunction(),local_predicate.getFunction() );
487  iterator = space->entitySet()->entityIterator( entity_dim, predicate );
488  }
489 
490  // Extract the coordinates and support ids of the nodes.
491  int local_num_node = iterator.size();
492  centers = Teuchos::ArrayRCP<double>( DIM*local_num_node);
493  support_ids = Teuchos::ArrayRCP<GO>( local_num_node );
494  Teuchos::Array<SupportId> node_supports;
495  EntityIterator begin = iterator.begin();
496  EntityIterator end = iterator.end();
497  int entity_counter = 0;
498  for ( EntityIterator entity = begin;
499  entity != end;
500  ++entity, ++entity_counter )
501  {
502  space->shapeFunction()->entitySupportIds(
503  *entity, node_supports );
504  DTK_CHECK( 1 == node_supports.size() );
505  support_ids[entity_counter] = node_supports[0];
506  space->localMap()->centroid(
507  *entity, centers(DIM*entity_counter,DIM) );
508  }
509 }
