59 ConsistentInterpolationOperator::ConsistentInterpolationOperator(
60  const Teuchos::RCP<const TpetraMap>& domain_map,
61  const Teuchos::RCP<const TpetraMap>& range_map,
62  const Teuchos::ParameterList& parameters )
63  : Base( domain_map, range_map )
64  , d_range_entity_dim( 0 )
65  , d_keep_missed_sol( false )
66  , d_missed_range_entity_ids( 0 )
67 {
68  // Get the topological dimension of the range entities.
69  Teuchos::ParameterList map_list =
70  parameters.sublist( "Consistent Interpolation" );
71  if ( map_list.isParameter("Range Entity Dimension") )
72  {
73  d_range_entity_dim = map_list.get<int>("Range Entity Dimension");
74  }
75 
76  // Get the search list.
77  d_search_list = parameters.sublist( "Search" );
78 
79  // If we want to keep the range data when we miss entities instead of
80  // zeros, turn this on.
81  if ( map_list.isParameter("Keep Missed Range Data") )
82  {
83  d_keep_missed_sol = map_list.get<bool>("Keep Missed Range Data");
84  if ( d_keep_missed_sol )
85  {
86  d_search_list.set("Track Missed Range Entities",true);
87  }
88  }

91 //---------------------------------------------------------------------------//
92 // Setup the map operator.
93 void ConsistentInterpolationOperator::setupImpl(
94  const Teuchos::RCP<FunctionSpace>& domain_space,
95  const Teuchos::RCP<FunctionSpace>& range_space )
96 {
97  DTK_REQUIRE( Teuchos::nonnull(domain_space) );
98  DTK_REQUIRE( Teuchos::nonnull(range_space) );
99 
100  // Extract the Support maps.
101  const Teuchos::RCP<const typename Base::TpetraMap> domain_map
102  = this->getDomainMap();
103  const Teuchos::RCP<const typename Base::TpetraMap> range_map
104  = this->getRangeMap();
105 
106  // Get the parallel communicator.
107  Teuchos::RCP<const Teuchos::Comm<int> > comm = domain_map->getComm();
108 
109  // Determine if we have range and domain data on this process.
110  bool nonnull_domain = Teuchos::nonnull( domain_space->entitySet() );
111  bool nonnull_range = Teuchos::nonnull( range_space->entitySet() );
112 
113  // Get the physical dimension.
114  int physical_dimension = 0;
115  if ( nonnull_domain )
116  {
117  physical_dimension = domain_space->entitySet()->physicalDimension();
118  }
119  else if ( nonnull_range )
120  {
121  physical_dimension = range_space->entitySet()->physicalDimension();
122  }
123 
124  // Get an iterator over the domain entities.
125  EntityIterator domain_iterator;
126  if ( nonnull_domain )
127  {
128  LocalEntityPredicate local_predicate(
129  domain_space->entitySet()->communicator()->getRank() );
130  PredicateFunction domain_predicate =
131  PredicateComposition::And(
132  domain_space->selectFunction(), local_predicate.getFunction() );
133  domain_iterator = domain_space->entitySet()->entityIterator(
134  domain_space->entitySet()->physicalDimension(),
135  domain_predicate );
136  }
137 
138  // Build a parallel search over the domain.
139  ParallelSearch psearch( comm,
140  physical_dimension,
141  domain_iterator,
142  domain_space->localMap(),
143  d_search_list );
144 
145  // Get an iterator over the range entities.
146  EntityIterator range_iterator;
147  if ( nonnull_range )
148  {
149  LocalEntityPredicate local_predicate(
150  range_space->entitySet()->communicator()->getRank() );
151  PredicateFunction range_predicate =
152  PredicateComposition::And(
153  range_space->selectFunction(), local_predicate.getFunction() );
154  range_iterator = range_space->entitySet()->entityIterator(
155  d_range_entity_dim, range_predicate );
156  }
157 
158  // Search the domain with the range.
159  psearch.search( range_iterator, range_space->localMap(), d_search_list );
160 
161  // If we are keeping track of range entities that were not mapped, extract
162  // them.
163  d_missed_range_entity_ids =
164  Teuchos::Array<EntityId>( psearch.getMissedRangeEntityIds() );
165 
166  // Determine the Support ids for the range entities found in the local domain
167  // on this process and the number of domain entities they were found in
168  // globally for averaging.
169  std::unordered_map<EntityId,GO> range_support_id_map;
170  Teuchos::RCP<Tpetra::Vector<double,int,SupportId> > scale_vector =
171  Tpetra::createVector<double,int,SupportId>( range_map );
172  {
173  // Extract the set of local range entities that were found in domain
174  // entities.
175  Teuchos::Array<int> export_ranks;
176  Teuchos::Array<GO> export_data;
177  Teuchos::Array<EntityId> domain_ids;
178  Teuchos::Array<EntityId>::const_iterator domain_id_it;
179  Teuchos::Array<GO> range_support_ids;
180  EntityIterator range_it;
181  EntityIterator range_begin = range_iterator.begin();
182  EntityIterator range_end = range_iterator.end();
183  for ( range_it = range_begin;
184  range_it != range_end;
185  ++range_it )
186  {
187  // Get the support id for the range entity.
188  range_space->shapeFunction()->entitySupportIds(
189  *range_it, range_support_ids );
190  DTK_CHECK( 1 == range_support_ids.size() );
191 
192  // Get the domain entities in which the range entity was found.
193  psearch.getDomainEntitiesFromRange( range_it->id(), domain_ids );
194 
195  // Add a scale factor for this range entity to the scaling vector.
196  DTK_CHECK( range_map->isNodeGlobalElement(range_support_ids[0]) );
197  scale_vector->replaceGlobalValue( range_support_ids[0],
198  1.0 / domain_ids.size() );
199 
200  // For each supporting domain entity, pair the range entity id and
201  // its support id.
202  for ( domain_id_it = domain_ids.begin();
203  domain_id_it != domain_ids.end();
204  ++domain_id_it )
205  {
206  export_ranks.push_back(
207  psearch.domainEntityOwnerRank(*domain_id_it) );
208 
209  export_data.push_back( range_support_ids[0] );
210  export_data.push_back( Teuchos::as<GO>(range_it->id()) );
211  }
212  }
213 
214  // Communicate the range entity Support data back to the domain parallel
215  // decomposition.
216  Tpetra::Distributor range_to_domain_dist( comm );
217  int num_import = range_to_domain_dist.createFromSends( export_ranks() );
218  Teuchos::Array<GO> import_data( 2*num_import );
219  Teuchos::ArrayView<const GO> export_data_view =
220  export_data();
221  range_to_domain_dist.doPostsAndWaits( export_data_view,
222  2,
223  import_data() );
224 
225  // Map the range entities to their support ids.
226  for ( int i = 0; i < num_import; ++i )
227  {
228  range_support_id_map.emplace(
229  Teuchos::as<EntityId>(import_data[2*i+1]),
230  import_data[2*i] );
231  }
232  }
233 
234  // Allocate the coupling matrix.
235  d_coupling_matrix =
236  Tpetra::createCrsMatrix<double,LO,GO>( range_map );
237 
238  // Construct the entries of the coupling matrix.
239  Teuchos::Array<EntityId> range_entity_ids;
240  Teuchos::Array<EntityId>::const_iterator range_entity_id_it;
241  Teuchos::ArrayView<const double> range_parametric_coords;
242  Teuchos::Array<double> domain_shape_values;
243  Teuchos::Array<double>::iterator domain_shape_it;
244  Teuchos::Array<GO> domain_support_ids;
245  EntityIterator domain_it;
246  EntityIterator domain_begin = domain_iterator.begin();
247  EntityIterator domain_end = domain_iterator.end();
248  for ( domain_it = domain_begin; domain_it != domain_end; ++domain_it )
249  {
250  // Get the domain Support ids supporting the domain entity.
251  domain_space->shapeFunction()->entitySupportIds(
252  *domain_it, domain_support_ids );
253 
254  // Get the range entities that mapped into this domain entity.
255  psearch.getRangeEntitiesFromDomain( domain_it->id(), range_entity_ids );
256 
257  // Sum into the global coupling matrix row for each domain.
258  for ( range_entity_id_it = range_entity_ids.begin();
259  range_entity_id_it != range_entity_ids.end();
260  ++range_entity_id_it )
261  {
262  // Get the parametric coordinates of the range entity in the
263  // domain entity.
264  psearch.rangeParametricCoordinatesInDomain(
265  domain_it->id(), *range_entity_id_it, range_parametric_coords );
266 
267  // Evaluate the shape function at the coordinates.
268  domain_space->shapeFunction()->evaluateValue(
269  *domain_it, range_parametric_coords, domain_shape_values );
270  DTK_CHECK( domain_shape_values.size() == domain_support_ids.size() );
271 
272  // Consistent interpolation requires one support location per
273  // range entity. Load the row for this range support location into
274  // the matrix.
275  DTK_CHECK( range_support_id_map.count(*range_entity_id_it) );
276  d_coupling_matrix->insertGlobalValues(
277  range_support_id_map.find(*range_entity_id_it)->second,
278  domain_support_ids(),
279  domain_shape_values() );
280  }
281  }
282 
283  // Finalize the coupling matrix.
284  d_coupling_matrix->fillComplete( domain_map, range_map );
285 
286  // Left-scale the matrix with the number of domain entities in which each
287  // range entity was found.
288  d_coupling_matrix->leftScale( *scale_vector );
289 
290  // If we want to keep the range data when we miss points, create the
291  // scaling vector.
292  if ( d_keep_missed_sol )
293  {
294  d_keep_range_vec = Tpetra::createVector<double,LO,GO>( range_map );
295  for ( auto& m : d_missed_range_entity_ids )
296  {
297  d_keep_range_vec->replaceGlobalValue( m, 1.0 );
298  }
299  }
300 }

302 //---------------------------------------------------------------------------//
303 // Apply the operator.
304 void ConsistentInterpolationOperator::applyImpl(
305  const TpetraMultiVector& X,
306  TpetraMultiVector &Y,
307  Teuchos::ETransp mode,
308  double alpha,
309  double beta ) const
310 {
311  // If we want to keep the range data when we miss points, make a work vec
312  // and get the parts we will zero out. Beta must be zero or the interface
313  // is violated.
314  Teuchos::RCP<Tpetra::Vector<Scalar,LO,GO> > work_vec;
315  if ( d_keep_missed_sol )
316  {
317  DTK_REQUIRE( 0.0 == beta );
318  DTK_REQUIRE( Teuchos::nonnull(d_keep_range_vec) );
319  work_vec = Tpetra::createVector<double,LO,GO>( this->getRangeMap() );
320  work_vec->elementWiseMultiply( 1.0, *d_keep_range_vec, Y, 0.0 );
321  }
322 
323  // Apply the coupling matrix.
324  d_coupling_matrix->apply( X, Y, mode, alpha, beta );
325 
326  // If we want to keep the range data when we miss points, add back in the
327  // components that got zeroed out.
328  if ( d_keep_missed_sol )
329  {
330  Y.update( 1.0, *work_vec, 1.0 );
331  }
332 }
