62 // Constructor.
63 template<class Basis,int DIM>
64 MovingLeastSquareReconstructionOperator<Basis,DIM>::
65 MovingLeastSquareReconstructionOperator(
66  const Teuchos::RCP<const TpetraMap>& domain_map,
67  const Teuchos::RCP<const TpetraMap>& range_map,
68  const Teuchos::ParameterList& parameters )
69  : Base( domain_map, range_map )
70  , d_use_knn( false )
71  , d_knn( 0 )
72  , d_radius( 0.0 )
73  , d_domain_entity_dim( 0 )
74  , d_range_entity_dim( 0 )
75 {
76  // Determine if we are doing kNN search or radius search.
77  if( parameters.isParameter("Type of Search") )
78  {
79  if ( "Radius" == parameters.get<std::string>("Type of Search") )
80  {
81  d_use_knn = false;
82  }
83  else if ( "Nearest Neighbor" == parameters.get<std::string>("Type of Search") )
84  {
85  d_use_knn = true;
86  }
87  else
88  {
89  // Otherwise we got an invalid search type.
90  DTK_INSIST( false );
91  }
92  }
93 
94  // If we are doing kNN support get the number of neighbors.
95  if( d_use_knn )
96  {
97  DTK_REQUIRE( parameters.isParameter("Num Neighbors") );
98  d_knn = parameters.get<int>("Num Neighbors");
99  }
100 
101  // Otherwise we are doing the radius search so get the basis radius.
102  else
103  {
104  DTK_REQUIRE( parameters.isParameter("RBF Radius") );
105  d_radius = parameters.get<double>("RBF Radius");
106  }
107 
108  // Get the topological dimension of the domain and range entities. This
109  // map will use their centroids for the point cloud.
110  if ( parameters.isParameter("Domain Entity Dimension") )
111  {
112  d_domain_entity_dim = parameters.get<int>("Domain Entity Dimension");
113  }
114  if ( parameters.isParameter("Range Entity Dimension") )
115  {
116  d_range_entity_dim = parameters.get<int>("Range Entity Dimension");
117  }
118 }

121 // Setup the map operator.
122 template<class Basis,int DIM>
123 void MovingLeastSquareReconstructionOperator<Basis,DIM>::setupImpl(
124  const Teuchos::RCP<FunctionSpace>& domain_space,
125  const Teuchos::RCP<FunctionSpace>& range_space )
126 {
127  DTK_REQUIRE( Teuchos::nonnull(domain_space) );
128  DTK_REQUIRE( Teuchos::nonnull(range_space) );
129 
130  // Extract the Support maps.
131  const Teuchos::RCP<const typename Base::TpetraMap> domain_map
132  = this->getDomainMap();
133  const Teuchos::RCP<const typename Base::TpetraMap> range_map
134  = this->getRangeMap();
135 
136  // Get the parallel communicator.
137  Teuchos::RCP<const Teuchos::Comm<int> > comm = domain_map->getComm();
138 
139  // Determine if we have range and domain data on this process.
140  bool nonnull_domain = Teuchos::nonnull( domain_space->entitySet() );
141  bool nonnull_range = Teuchos::nonnull( range_space->entitySet() );
142 
143  // Extract the source nodes and their ids.
144  Teuchos::ArrayRCP<double> source_centers;
145  Teuchos::ArrayRCP<GO> source_support_ids;
146  getNodeCoordsAndIds( domain_space, d_domain_entity_dim,
147  source_centers, source_support_ids );
148 
149  // Extract the target nodes and their ids.
150  Teuchos::ArrayRCP<double> target_centers;
151  Teuchos::ArrayRCP<GO> target_support_ids;
152  getNodeCoordsAndIds( range_space, d_range_entity_dim,
153  target_centers, target_support_ids );
154 
155  // Calculate an approximate neighborhood distance for the local target
156  // centers. If using kNN, compute an approximation. If doing a radial
157  // search, use the radius. We will use these distances to expand the local
158  // bounding box to ensure we find all of our neighbors in parallel.
159  double target_proximity = 0.0;
160  if ( d_use_knn )
161  {
162  // Get the local bounding box.
163  Teuchos::Tuple<double,6> local_box;
164  range_space->entitySet()->localBoundingBox( local_box );
165 
166  // Calculate the largest span of the cardinal directions.
167  target_proximity = local_box[3] - local_box[0];
168  for ( int d = 1; d < DIM; ++d )
169  {
170  target_proximity = std::max( target_proximity,
171  local_box[d+3] - local_box[d] );
172  }
173 
174  // Take the proximity to be 10% of the largest distance.
175  target_proximity *= 0.1;
176  }
177  else
178  {
179  target_proximity = d_radius;
180  }
181 
182  // Gather the source centers that are in the proximity of the target
183  // centers on this proc.
184  Teuchos::Array<double> dist_sources;
185  CenterDistributor<DIM> distributor(
186  comm, source_centers(), target_centers(), target_proximity, dist_sources );
187 
188  // Gather the global ids of the source centers that are within the proximity of
189  // the target centers on this proc.
190  Teuchos::Array<GO> dist_source_support_ids( distributor.getNumImports() );
191  Teuchos::ArrayView<const GO> source_support_ids_view = source_support_ids();
192  distributor.distribute( source_support_ids_view, dist_source_support_ids() );
193 
194  // Build the source/target pairings.
195  SplineInterpolationPairing<DIM> pairings(
196  dist_sources, target_centers(), d_use_knn, d_knn, d_radius );
197 
198  // Build the basis.
199  Teuchos::RCP<Basis> basis = BP::create();
200 
201  // Build the interpolation matrix.
202  Teuchos::ArrayRCP<SupportId> children_per_parent =
203  pairings.childrenPerParent();
204  SupportId max_entries_per_row = *std::max_element(
205  children_per_parent.begin(), children_per_parent.end() );
206  d_coupling_matrix = Teuchos::rcp( new Tpetra::CrsMatrix<Scalar,LO,GO>(
207  range_map,
208  max_entries_per_row) );
209  Teuchos::ArrayView<const double> target_view;
210  Teuchos::Array<GO> indices( max_entries_per_row );
211  Teuchos::ArrayView<const double> values;
212  Teuchos::ArrayView<const unsigned> pair_gids;
213  int nn = 0;
214  int local_num_tgt = target_support_ids.size();
215  for ( int i = 0; i < local_num_tgt; ++i )
216  {
217  // If there is no support for this target center then do not build a
218  // local basis.
219  if ( 0 < pairings.childCenterIds(i).size() )
220  {
221  // Get a view of this target center.
222  target_view = target_centers(i*DIM,DIM);
223 
224  // Build the local interpolation problem.
225  LocalMLSProblem<Basis,DIM> local_problem(
226  target_view, pairings.childCenterIds(i),
227  dist_sources, *basis, pairings.parentSupportRadius(i) );
228 
229  // Get MLS shape function values for this target point.
230  values = local_problem.shapeFunction();
231  nn = values.size();
232 
233  // Populate the interpolation matrix row.
234  pair_gids = pairings.childCenterIds(i);
235  for ( int j = 0; j < nn; ++j )
236  {
237  indices[j] = dist_source_support_ids[ pair_gids[j] ];
238  }
239  d_coupling_matrix->insertGlobalValues(
240  target_support_ids[i], indices(0,nn), values );
241  }
242  }
243  d_coupling_matrix->fillComplete( domain_map, range_map );
244  DTK_ENSURE( d_coupling_matrix->isFillComplete() );
245 }

269 //---------------------------------------------------------------------------//
270 // Extract node coordinates and ids from an iterator.
271 template<class Basis,int DIM>
272 void MovingLeastSquareReconstructionOperator<Basis,DIM>::getNodeCoordsAndIds(
273  const Teuchos::RCP<FunctionSpace>& space,
274  const int entity_dim,
275  Teuchos::ArrayRCP<double>& centers,
276  Teuchos::ArrayRCP<GO>& support_ids ) const
277 {
278  // Get an iterator over the local nodes.
279  EntityIterator iterator;
280  if ( Teuchos::nonnull(space->entitySet()) )
281  {
282  LocalEntityPredicate local_predicate(
283  space->entitySet()->communicator()->getRank() );
284  PredicateFunction predicate =
285  PredicateComposition::And(
286  space->selectFunction(),local_predicate.getFunction() );
287  iterator = space->entitySet()->entityIterator( entity_dim, predicate );
288  }
289 
290  // Extract the coordinates and support ids of the nodes.
291  int local_num_node = iterator.size();
292  centers = Teuchos::ArrayRCP<double>( DIM*local_num_node);
293  support_ids = Teuchos::ArrayRCP<GO>( local_num_node );
294  Teuchos::Array<SupportId> node_supports;
295  EntityIterator begin = iterator.begin();
296  EntityIterator end = iterator.end();
297  int entity_counter = 0;
298  for ( EntityIterator entity = begin;
299  entity != end;
300  ++entity, ++entity_counter )
301  {
302  space->shapeFunction()->entitySupportIds(
303  *entity, node_supports );
304  DTK_CHECK( 1 == node_supports.size() );
305  support_ids[entity_counter] = node_supports[0];
306  space->localMap()->centroid(
307  *entity, centers(DIM*entity_counter,DIM) );
308  }
309 }
