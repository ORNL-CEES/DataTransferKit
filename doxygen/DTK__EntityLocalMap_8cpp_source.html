60 // Perform a safeguard check for mapping a point to the reference space
61 // of an entity using the given tolerance. Default implementation checks if
62 // the point is inside the bounding box of the entity.
63 bool EntityLocalMap::isSafeToMapToReferenceFrame(
64  const Entity& entity,
65  const Teuchos::ArrayView<const double>& point ) const
66 {
67  // Get the bounding box of the entity.
68  Teuchos::Tuple<double,6> entity_box;
69  entity.boundingBox( entity_box );
70 
71  // Check if the point is in the bounding box of the entity.
72  double tolerance = 1.0e-6;
73  int space_dim = entity.physicalDimension();
74  bool in_x = true;
75  if ( space_dim > 0 )
76  {
77  double x_tol = (entity_box[3] - entity_box[0])*tolerance;
78  in_x = ( (point[0] >= (entity_box[0] - x_tol)) &&
79  (point[0] <= (entity_box[3] + x_tol)) );
80  }
81  bool in_y = true;
82  if ( space_dim > 1 )
83  {
84  double y_tol = (entity_box[4] - entity_box[1])*tolerance;
85  in_y = ( (point[1] >= (entity_box[1] - y_tol)) &&
86  (point[1] <= (entity_box[4] + y_tol)) );
87  }
88  bool in_z = true;
89  if ( space_dim > 2 )
90  {
91  double z_tol = (entity_box[5] - entity_box[2])*tolerance;
92  in_z = ( (point[2] >= (entity_box[2] - z_tol)) &&
93  (point[2] <= (entity_box[5] + z_tol)) );
94  }
95  return (in_x && in_y && in_z);
96 }

99 // Compute the normal on a face (3D) or edge (2D) at a given reference point.
100 void EntityLocalMap::normalAtReferencePoint(
101  const Entity& entity,
102  const Entity& parent_entity,
103  const Teuchos::ArrayView<const double>& reference_point,
104  const Teuchos::ArrayView<double>& normal ) const
105 {
106  // Determine the reference dimension.
107  int physical_dim = entity.physicalDimension();
108  int ref_dim = physical_dim - 1;
109 
110  // Create a perturbation.
111  double perturbation =
112  std::sqrt( std::numeric_limits<double>::epsilon() );
113 
114  // 3D/face case.
115  if ( 2 == ref_dim )
116  {
117  DTK_CHECK( 3 == reference_point.size() );
118  DTK_CHECK( 3 == normal.size() );
119 
120  // Create extra points.
121  Teuchos::Array<double> ref_p1( reference_point );
122  Teuchos::Array<double> ref_p2( reference_point );
123 
124  // Apply a perturbation to the extra points.
125  double p1_sign = 1.0;
126  ref_p1[0] += perturbation;
127  if ( !this->checkPointInclusion(entity,ref_p1()) )
128  {
129  ref_p1[0] -= 2*perturbation;
130  p1_sign = -1.0;
131  }
132  double p2_sign = 1.0;
133  ref_p2[1] += perturbation;
134  if ( !this->checkPointInclusion(entity,ref_p2()) )
135  {
136  ref_p2[1] -= 2*perturbation;
137  p2_sign = -1.0;
138  }
139 
140  // Map the perturbed points to the physical frame.
141  Teuchos::Array<double> p0( physical_dim );
142  this->mapToPhysicalFrame( entity, reference_point(), p0() );
143  Teuchos::Array<double> p1( physical_dim );
144  this->mapToPhysicalFrame( entity, ref_p1(), p1() );
145  Teuchos::Array<double> p2( physical_dim );
146  this->mapToPhysicalFrame( entity, ref_p2(), p2() );
147 
148  // Compute the cross product of the tangents produced by the
149  // perturbation.
150  Teuchos::Array<double> tan1( physical_dim );
151  Teuchos::Array<double> tan2( physical_dim );
152  for ( int d = 0; d < physical_dim; ++d )
153  {
154  tan1[d] = p1_sign*(p1[d] - p0[d]);
155  tan2[d] = p2_sign*(p2[d] - p0[d]);
156  }
157  normal[0] = tan1[1]*tan2[2] - tan1[2]*tan2[1];
158  normal[1] = tan1[2]*tan2[0] - tan1[0]*tan2[2];
159  normal[2] = tan1[0]*tan2[1] - tan1[1]*tan2[0];
160  }
161 
162  // 2D/edge case.
163  else if ( 1 == ref_dim )
164  {
165  DTK_CHECK( 2 == reference_point.size() );
166  DTK_CHECK( 2 == normal.size() );
167 
168  // Create extra points.
169  Teuchos::Array<double> ref_p1( reference_point );
170 
171  // Apply a perturbation to the extra points.
172  double p1_sign = 1.0;
173  ref_p1[0] += perturbation;
174  if ( !this->checkPointInclusion(entity,ref_p1()) )
175  {
176  ref_p1[0] -= 2*perturbation;
177  p1_sign = -1.0;
178  }
179 
180  // Map the perturbed points to the physical frame.
181  Teuchos::Array<double> p0( physical_dim );
182  this->mapToPhysicalFrame( entity, reference_point(), p0() );
183  Teuchos::Array<double> p1( physical_dim );
184  this->mapToPhysicalFrame( entity, ref_p1(), p1() );
185 
186  // Compute the cross product of the tangents produced by the
187  // perturbation.
188  Teuchos::Array<double> tan( physical_dim );
189  for ( int d = 0; d < physical_dim; ++d )
190  {
191  tan[d] = p1_sign*(p1[d] - p0[d]);
192  }
193  normal[0] = -tan[0];
194  normal[1] = tan[1];
195  }
196 
197  // Normalize the normal vector.
198  double norm = 0.0;
199  for ( int d = 0; d < physical_dim; ++d )
200  {
201  norm += normal[d]*normal[d];
202  }
203  norm = std::sqrt(norm);
204  for ( int d = 0; d < physical_dim; ++d )
205  {
206  normal[d] /= norm;
207  }
208 }
