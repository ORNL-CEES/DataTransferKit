101 //---------------------------------------------------------------------------//
102 // Return the Cartesian bounding box around an entity.
103 void STKMeshEntityImpl::boundingBox( Teuchos::Tuple<double,6>& bounds ) const
104 {
105  DTK_REQUIRE( Teuchos::nonnull(d_bulk_data) );
106 
107  Intrepid::FieldContainer<double> node_coords =
108  STKMeshHelpers::getEntityNodeCoordinates(
109  Teuchos::Array<stk::mesh::Entity>(1,d_extra_data->d_stk_entity),
110  *d_bulk_data );
111  DTK_CHECK( node_coords.rank() == 3 );
112  DTK_CHECK( node_coords.dimension(0) == 1 );
113 
114  double max = std::numeric_limits<double>::max();
115  bounds = Teuchos::tuple( max, max, max, -max, -max, -max );
116  int space_dim = node_coords.dimension(2);
117  for ( int n = 0; n < node_coords.dimension(1); ++n )
118  {
119  for ( int d = 0; d < space_dim; ++d )
120  {
121  bounds[d] = std::min( bounds[d], node_coords(0,n,d) );
122  bounds[d+3] = std::max( bounds[d+3], node_coords(0,n,d) );
123  }
124  }
125  for ( int d = space_dim; d < 3; ++d )
126  {
127  bounds[d] = -max;
128  bounds[d+3] = max;
129  }
130 
131 }

134 // Determine if an entity is in the block with the given id.
135 bool STKMeshEntityImpl::inBlock( const int block_id ) const
136 {
137  DTK_REQUIRE( Teuchos::nonnull(d_bulk_data) );
138  const stk::mesh::PartVector& all_parts =
139  d_bulk_data->mesh_meta_data().get_parts();
140  stk::mesh::Bucket& entity_bucket =
141  d_bulk_data->bucket( d_extra_data->d_stk_entity );
142  for ( auto part_it = all_parts.begin();
143  part_it != all_parts.end();
144  ++part_it )
145  {
146  if ( Teuchos::as<int>((*part_it)->mesh_meta_data_ordinal()) == block_id )
147  {
148  return entity_bucket.member( **part_it );
149  }
150  }
151  return false;
152 }
