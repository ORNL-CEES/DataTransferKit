211 //---------------------------------------------------------------------------//
212 // Increment the iterator implementation forward until either a valid
213 // increment is found or we have reached the end.
214 void EntityIterator::increment()
215 {
216  DTK_REQUIRE( b_iterator_impl );
217  DTK_REQUIRE( *b_iterator_impl != b_iterator_impl->end() );
218 
219  // Apply the increment operator.
220  EntityIterator& it = b_iterator_impl->operator++();
221 
222  // Get the end of the range.
223  EntityIterator end = b_iterator_impl->end();
224 
225  // If the we are not at the end or the predicate is not satisfied by the
226  // current element, increment until either of these conditions is
227  // satisfied.
228  while ( it != end && !b_predicate(*it) )
229  {
230  it = b_iterator_impl->operator++();
231  }
232 }
