56 IntrepidSideCell::IntrepidSideCell(

79 void IntrepidSideCell::updateCellState()
80 {
81  unsigned space_dim = d_parent_topology.getDimension();
82 
83  // Compute the Jacobian.
84  Intrepid::CellTools<Scalar>::setJacobian(
85  this->d_jacobian, this->d_cub_points,
86  this->d_cell_node_coords, d_parent_topology );
87 
88  // Compute the cell side measures.
89  switch ( space_dim )
90  {
91  case 3:
92  {
93  // Face case.
94  Intrepid::FunctionSpaceTools::computeFaceMeasure<Scalar>(
95  this->d_weighted_measures, this->d_jacobian,
96  this->d_cub_weights, d_side_id, d_parent_topology );
97  }
98  break;
99 
100  case 2:
101  {
102  // Edge case.
103  Intrepid::FunctionSpaceTools::computeEdgeMeasure<Scalar>(
104  this->d_weighted_measures, this->d_jacobian,
105  this->d_cub_weights, d_side_id, d_parent_topology );
106  }
107  break;
108 
109  default:
110  DTK_INSIST( 3 == space_dim || 2 == space_dim );
111  break;
112  }
113 
114  // Compute physical frame integration point coordinates.
115  Intrepid::CellTools<Scalar>::mapToPhysicalFrame(
116  this->d_physical_ip_coordinates, this->d_cub_points,
117  this->d_cell_node_coords, d_parent_topology );
118 }

125 void IntrepidSideCell::mapToCellPhysicalFrame(
126  const MDArray& parametric_coords, MDArray& physical_coords )
127 {
128  DTK_REQUIRE( 2 == parametric_coords.rank() );
129  DTK_REQUIRE( 3 == physical_coords.rank() );
130  DTK_REQUIRE( parametric_coords.dimension(1) ==
131  Teuchos::as<int>(this->d_topology.getDimension()) );
132  DTK_REQUIRE( physical_coords.dimension(0) ==
133  this->d_cell_node_coords.dimension(0) );
134  DTK_REQUIRE( physical_coords.dimension(1) ==
135  parametric_coords.dimension(0) );
136  DTK_REQUIRE( physical_coords.dimension(2) ==
137  Teuchos::as<int>(d_parent_topology.getDimension()) );
138 
139  MDArray mapped_coords( parametric_coords.dimension(0),
140  d_parent_topology.getDimension() );
141 
142  Intrepid::CellTools<Scalar>::mapToReferenceSubcell(
143  mapped_coords, parametric_coords, this->d_topology.getDimension(),
144  d_side_id, d_parent_topology );
145 
146  Intrepid::CellTools<Scalar>::mapToPhysicalFrame(
147  physical_coords, mapped_coords,
148  this->d_cell_node_coords, d_parent_topology );
149 }
