58 IntrepidCell::IntrepidCell(
59  const shards::CellTopology& cell_topology, const unsigned degree )
60  : d_topology( cell_topology )
61  , d_cub_points(0,0)
62  , d_cub_weights(0)
63  , d_jacobian(0,0,0,0)
64  , d_jacobian_det(0,0)
65  , d_weighted_measures(0,0)
66  , d_physical_ip_coordinates(0,0,0)
67 {
68  Intrepid::DefaultCubatureFactory<Scalar,MDArray> cub_factory;
69  d_cubature = cub_factory.create( d_topology, degree );
70 
71  unsigned num_cub_points = d_cubature->getNumPoints();
72  unsigned cub_dim = cell_topology.getDimension();
73 
74  d_cub_points.resize( num_cub_points, cub_dim );
75  d_cub_weights.resize( num_cub_points );
76  d_cubature->getCubature( d_cub_points, d_cub_weights );
77 }

101 void IntrepidCell::allocateCellState( const MDArray& cell_node_coords )
102 {
103  // Store the cell node coords as the current state.
104  setCellNodeCoordinates( cell_node_coords );
105 
106  // Get required dimensions.
107  int num_cells = d_cell_node_coords.dimension(0);
108  int num_ip = d_cub_points.dimension(0);
109  int space_dim = d_cub_points.dimension(1);
110 
111  // Resize arrays.
112  d_jacobian.resize( num_cells, num_ip, space_dim, space_dim );
113  d_jacobian_det.resize( num_cells, num_ip );
114  d_weighted_measures.resize( num_cells, num_ip );
115  d_physical_ip_coordinates.resize( num_cells, num_ip, space_dim );
116 }

123 void IntrepidCell::updateCellState()
124 {
125  // Compute the Jacobian.
126  Intrepid::CellTools<Scalar>::setJacobian(
127  d_jacobian, d_cub_points, d_cell_node_coords, d_topology );
128 
129  // Compute the determinant of the Jacobian.
130  Intrepid::CellTools<Scalar>::setJacobianDet(
131  d_jacobian_det, d_jacobian );
132 
133  // Compute the cell measures.
134  Intrepid::FunctionSpaceTools::computeCellMeasure<Scalar>(
135  d_weighted_measures, d_jacobian_det, d_cub_weights );
136 
137  // Compute physical frame integration point coordinates.
138  Intrepid::CellTools<Scalar>::mapToPhysicalFrame(
139  d_physical_ip_coordinates, d_cub_points,
140  d_cell_node_coords, d_topology );
141 }

174 void IntrepidCell::mapToCellPhysicalFrame(
175  const MDArray& parametric_coords, MDArray& physical_coords )
176 {
177  DTK_REQUIRE( 2 == parametric_coords.rank() );
178  DTK_REQUIRE( 3 == physical_coords.rank() );
179  DTK_REQUIRE( parametric_coords.dimension(1) ==
180  Teuchos::as<int>(d_topology.getDimension()) );
181  DTK_REQUIRE( physical_coords.dimension(0) ==
182  d_cell_node_coords.dimension(0) );
183  DTK_REQUIRE( physical_coords.dimension(1) ==
184  parametric_coords.dimension(0) );
185  DTK_REQUIRE( physical_coords.dimension(2) ==
186  Teuchos::as<int>(d_topology.getDimension()) );
187 
188  Intrepid::CellTools<Scalar>::mapToPhysicalFrame(
189  physical_coords, parametric_coords,
190  d_cell_node_coords, d_topology );
191 }
