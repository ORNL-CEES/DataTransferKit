55 template<int DIM>
56 SplineInterpolationPairing<DIM>::SplineInterpolationPairing(
57  const Teuchos::ArrayView<const double>& child_centers,
58  const Teuchos::ArrayView<const double>& parent_centers,
59  const bool use_knn,
60  const unsigned num_neighbors,
61  const double radius )
62 {
63  DTK_REQUIRE( 0 == child_centers.size() % DIM );
64  DTK_REQUIRE( 0 == parent_centers.size() % DIM );
65 
66  // Setup the kD-tree
67  unsigned leaf_size = 30;
68  NanoflannTree<DIM> tree( child_centers, leaf_size );
69 
70  // Allocate arrays
71  unsigned num_parents = parent_centers.size() / DIM;
72  d_pairings.resize( num_parents );
73  d_pair_sizes = Teuchos::ArrayRCP<EntityId>( num_parents );
74  d_radii.resize( num_parents );
75 
76  // Search for pairs
77  for ( unsigned i = 0; i < num_parents; ++i )
78  {
79  // If kNN do the nearest neighbor search for kNN and calculate a
80  // radius. The radius will be a small fraction larger than the
81  // farthest neighbor. An alternative to this would be to find the
82  // kNN+1 neighbors and use the last neighbor's distance as the
83  // radius.
84  if ( use_knn )
85  {
86  // Get the knn neighbors
87  d_pairings[i] =
88  tree.nnSearch( parent_centers(DIM*i,DIM), num_neighbors );
89 
90  // Get the radius from kNN. Make it slightly larger so the last
91  // neighbor gives a non-zero contribution to the interpolant.
92  d_radii[i] =
93  EuclideanDistance<DIM>::distance(
94  parent_centers(DIM*i,DIM).getRawPtr(),
95  child_centers(DIM*d_pairings[i].back(),DIM).getRawPtr() );
96  d_radii[i] *= 1.01;
97  }
98 
99  // Otherwise do the radius search.
100  else
101  {
102  d_pairings[i] =
103  tree.radiusSearch( parent_centers(DIM*i,DIM), radius );
104  d_radii[i] = radius;
105  }
106 
107  // Get the size of the support.
108  d_pair_sizes[i] = d_pairings[i].size();
109  }
110 }
