60 // Get the local bounding box of entities of the set. Default implementation
61 // gathers the bounding boxes of local entities.
62 void EntitySet::localBoundingBox( Teuchos::Tuple<double,6>& bounds ) const
63 {
64  LocalEntityPredicate local_predicate( communicator()->getRank() );
65  PredicateFunction select_func = local_predicate.getFunction();
66  double max = std::numeric_limits<double>::max();
67  bounds = Teuchos::tuple( max, max, max, -max, -max, -max );
68  EntityIterator entity_begin;
69  EntityIterator entity_end;
70  EntityIterator entity_it;
71  EntityIterator dim_it;
72  Teuchos::Tuple<double,6> entity_bounds;
73  for ( int i = 0; i < 4; ++i )
74  {
75  dim_it = this->entityIterator( i, select_func );
76  entity_begin = dim_it.begin();
77  entity_end = dim_it.end();
78  for ( entity_it = entity_begin;
79  entity_it != entity_end;
80  ++entity_it )
81  {
82  entity_it->boundingBox( entity_bounds );
83  for ( int n = 0; n < 3; ++n )
84  {
85  bounds[n] = std::min( bounds[n], entity_bounds[n] );
86  bounds[n+3] = std::max( bounds[n+3], entity_bounds[n+3] );
87  }
88  }
89  }
90 }

93 // Get the global bounding box of entities of the set. Default implementation
94 // performs a parallel reduction using the local bounding boxes.
95 void EntitySet::globalBoundingBox( Teuchos::Tuple<double,6>& bounds ) const
96 {
97  double max = std::numeric_limits<double>::max();
98  bounds = Teuchos::tuple( max, max, max, -max, -max, -max );
99  Teuchos::Tuple<double,6> local_bounds;
100  this->localBoundingBox( local_bounds );
101  Teuchos::reduceAll( *(this->communicator()), Teuchos::REDUCE_MIN, 3,
102  &local_bounds[0], &bounds[0] );
103  Teuchos::reduceAll( *(this->communicator()), Teuchos::REDUCE_MAX, 3,
104  &local_bounds[3], &bounds[3] );
105 }
