63 template<class Basis,int DIM>
64 LocalMLSProblem<Basis,DIM>::LocalMLSProblem(
65  const Teuchos::ArrayView<const double>& target_center,
66  const Teuchos::ArrayView<const unsigned>& source_lids,
67  const Teuchos::ArrayView<const double>& source_centers,
68  const Basis& basis,
69  const double radius )
70  : d_shape_function( source_lids.size() )
71 {
72  DTK_REQUIRE( 0 == source_centers.size() % DIM );
73  DTK_REQUIRE( 0 == target_center.size() % DIM );
74 
75  // Number of source centers supporting this target center.
76  int num_sources = source_lids.size();
77  int poly_size = 0;
78  Teuchos::SerialDenseMatrix<int,double> P;
79  Teuchos::SerialDenseVector<int,double> target_poly;
80 
81  // Make Phi.
82  Teuchos::SerialDenseMatrix<int,double> phi( num_sources, num_sources );
83  Teuchos::ArrayView<const double> source_center_view;
84  double dist = 0.0;
85  for ( int i = 0; i < num_sources; ++i )
86  {
87  source_center_view = source_centers(DIM*source_lids[i],DIM);
88  dist = EuclideanDistance<DIM>::distance(
89  target_center.getRawPtr(), source_center_view.getRawPtr() );
90  phi(i,i) = BP::evaluateValue( basis, radius, dist );
91  }
92 
93  // Make P.
94  Teuchos::Array<int> poly_ids(1,0);
95  int poly_id = 0;
96  P.reshape( num_sources, poly_id+1 );
97  for ( int i = 0; i < num_sources; ++i )
98  {
99  source_center_view = source_centers(DIM*source_lids[i],DIM);
100  P(i,poly_id) = polynomialCoefficient( 0, source_center_view );
101  }
102  ++poly_id;
103 
104  // Add polynomial columns until we are full rank.
105  bool full_rank = false;
106  int num_poly = 10;
107  int total_poly = std::min( num_poly, num_sources );
108  for ( int j = 1; j < total_poly; ++j )
109  {
110  // Add the next column.
111  P.reshape( num_sources, poly_id+1 );
112  for ( int i = 0; i < num_sources; ++i )
113  {
114  source_center_view = source_centers(DIM*source_lids[i],DIM);
115  P(i,poly_id) = polynomialCoefficient( j, source_center_view );
116  }
117 
118  // Check for rank deficiency.
119  full_rank = isFullRank( P );
120 
121  // If we are full rank, add this coefficient.
122  if ( full_rank )
123  {
124  poly_ids.push_back( j );
125  ++poly_id;
126  }
127 
128  // If we are rank deficient, remove the last column.
129  else
130  {
131  P.reshape( num_sources, poly_id );
132  }
133  }
134 
135  // Make p.
136  poly_size = poly_ids.size();
137  target_poly.resize( poly_size );
138  for ( int i = 0; i < poly_size; ++i )
139  {
140  target_poly(i) = polynomialCoefficient( poly_ids[i], target_center );
141  }
142 
143  // Construct b.
144  Teuchos::SerialDenseMatrix<int,double> b( poly_size, num_sources );
145  b.multiply( Teuchos::TRANS, Teuchos::NO_TRANS, 1.0, P, phi, 0.0 );
146 
147  // Construct the A matrix.
148  Teuchos::SerialDenseMatrix<int,double> A( poly_size, poly_size );
149  {
150  // Build A.
151  Teuchos::SerialDenseMatrix<int,double> work( num_sources, poly_size );
152  work.multiply( Teuchos::NO_TRANS, Teuchos::NO_TRANS, 1.0, phi, P, 0.0 );
153  A.multiply( Teuchos::TRANS, Teuchos::NO_TRANS, 1.0, P, work, 0.0 );
154  }
155 
156  // Apply the inverse of the A matrix to b.
157  Teuchos::LAPACK<int,double> lapack;
158  double A_rcond = std::numeric_limits<double>::epsilon();
159  Teuchos::Array<double> work( 4 * A.numCols() );
160  Teuchos::SerialDenseVector<int,double> s( poly_size );
161  int rank = 0;
162  int info = 0;
163 
164  // Estimate the reciprocal of the condition number.
165  Teuchos::Array<int> ipiv( std::min(A.numRows(),A.numCols()) );
166  Teuchos::SerialDenseMatrix<int,double> LU_A( A );
167  lapack.GETRF( LU_A.numRows(), LU_A.numCols(), LU_A.values(),
168  LU_A.numRows(), ipiv.getRawPtr(), &info );
169  DTK_CHECK( 0 == info );
170 
171  Teuchos::Array<int> iwork( A.numCols() );
172  lapack.GECON( '1', LU_A.numCols(), LU_A.values(),
173  LU_A.numRows(), A.normOne(), &A_rcond,
174  work.getRawPtr(), iwork.getRawPtr(), &info );
175  DTK_CHECK( 0 == info );
176 
177  // Get the optimal work size.
178  lapack.GELSS( A.numRows(), A.numCols(), b.numCols(),
179  A.values(), A.numRows(),
180  b.values(), b.numRows(), s.values(),
181  A_rcond, &rank, work.getRawPtr(), -1, &info );
182  DTK_CHECK( 0 == info );
183 
184  // Apply the inverse of A to b.
185  work.resize( work[0] );
186  lapack.GELSS( A.numRows(), A.numCols(), b.numCols(),
187  A.values(), A.numRows(),
188  b.values(), b.numRows(), s.values(),
189  A_rcond, &rank, work.getRawPtr(), work.size(), &info );
190  DTK_CHECK( 0 == info );
191 
192  // Construct the basis.
193  Teuchos::SerialDenseMatrix<int,double> shape_matrix(
194  Teuchos::View, d_shape_function.getRawPtr(),
195  1, 1, d_shape_function.size() );
196  shape_matrix.multiply( Teuchos::TRANS, Teuchos::NO_TRANS,
197  1.0, target_poly, b, 0.0 );
198 }

201 // Get a polynomial coefficient.
202 template<class Basis,int DIM>
203 double LocalMLSProblem<Basis,DIM>::polynomialCoefficient(
204  const int coeff, const Teuchos::ArrayView<const double>& center ) const
205 {
206  switch( coeff )
207  {
208  // Linear.
209  case( 0 ):
210  return 1.0;
211  break;
212  case( 1 ):
213  return center[0];
214  break;
215  case( 2 ):
216  return center[1];
217  break;
218  case( 3 ):
219  return center[2];
220  break;
221 
222  // Quadratic
223  case( 4 ):
224  return center[0]*center[1];
225  break;
226  case( 5 ):
227  return center[0]*center[2];
228  break;
229  case( 6 ):
230  return center[1]*center[2];
231  break;
232  case( 7 ):
233  return center[0]*center[0];
234  break;
235  case( 8 ):
236  return center[1]*center[1];
237  break;
238  case( 9 ):
239  return center[2]*center[2];
240  break;
241  }
242  return 0.0;
243 }

246 // Check if a matrix is full rank.
247 template<class Basis,int DIM>
248 bool LocalMLSProblem<Basis,DIM>::isFullRank(
249  const Teuchos::SerialDenseMatrix<int,double>& matrix ) const
250 {
251  // Copy the matrix.
252  Teuchos::SerialDenseMatrix<int,double> A = matrix;
253 
254  // Determine the full rank.
255  int full_rank = std::min( A.numRows(), A.numCols() );
256 
257  // Compute the singular value decomposition.
258  Teuchos::LAPACK<int,double> lapack;
259  Teuchos::Array<double> S( full_rank );
260  Teuchos::SerialDenseMatrix<int,double> U( A.numRows(), A.numRows() );
261  Teuchos::SerialDenseMatrix<int,double> VT( A.numCols(), A.numCols() );
262  Teuchos::Array<double> work( full_rank );
263  Teuchos::Array<double> rwork( full_rank );
264  int info = 0;
265  lapack.GESVD( 'A', 'A',
266  A.numRows(), A.numCols(), A.values(), A.numRows(),
267  S.getRawPtr(),
268  U.values(), U.numRows(),
269  VT.values(), VT.numRows(),
270  work.getRawPtr(), -1, rwork.getRawPtr(), &info );
271  DTK_CHECK( 0 == info );
272 
273  work.resize( work[0] );
274  rwork.resize( work.size() );
275  lapack.GESVD( 'A', 'A',
276  A.numRows(), A.numCols(), A.values(), A.numRows(),
277  S.getRawPtr(),
278  U.values(), U.numRows(),
279  VT.values(), VT.numRows(),
280  work.getRawPtr(), work.size(), rwork.getRawPtr(), &info );
281  DTK_CHECK( 0 == info );
282 
283  // Check the singular values. If they are greater than delta they count.
284  double epsilon = std::numeric_limits<double>::epsilon();
285  double delta = S[0]*epsilon;
286  int rank = std::count_if( S.begin(), S.end(),
287  [=](double s){ return (s > delta); } );
288 
289  return ( rank == full_rank );
290 }
