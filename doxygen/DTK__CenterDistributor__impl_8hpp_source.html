58 template<int DIM>
59 CenterDistributor<DIM>::CenterDistributor(
60  const Teuchos::RCP<const Teuchos::Comm<int> >& comm,
61  const Teuchos::ArrayView<const double>& source_centers,
62  const Teuchos::ArrayView<const double>& target_centers,
63  const double radius,
64  Teuchos::Array<double>& target_decomp_source_centers )
65  : d_distributor( new Tpetra::Distributor(comm) )
66 {
67  DTK_REQUIRE( 0 == source_centers.size() % DIM );
68  DTK_REQUIRE( 0 == target_centers.size() % DIM );
69 
70  // Build the import/export data.
71  Teuchos::Array<int> export_procs;
72  {
73  // Compute the radius to expand the local domain with.
74  double radius_tol = 1.0e-2;
75  double radius_expand = radius * ( 1.0 + radius_tol );
76 
77  // Gather the bounding domains for each target proc.
78  CloudDomain<DIM> local_target_domain =
79  localCloudDomain( target_centers );
80  local_target_domain.expand( radius_expand );
81  Teuchos::Array<CloudDomain<DIM> > global_target_domains(
82  comm->getSize() );
83  Teuchos::gatherAll<int,CloudDomain<DIM> >(
84  *comm,
85  1,
86  &local_target_domain,
87  global_target_domains.size(),
88  global_target_domains.getRawPtr() );
89 
90  // Get those that are neighbors to this source proc.
91  CloudDomain<DIM> local_source_domain =
92  localCloudDomain( source_centers );
93  Teuchos::Array<CloudDomain<DIM> > neighbor_target_domains;
94  Teuchos::Array<int> neighbor_ranks;
95  for ( unsigned i = 0; i < global_target_domains.size(); ++i )
96  {
97  if ( local_source_domain.checkForIntersection(
98  global_target_domains[i]) )
99  {
100  neighbor_target_domains.push_back(global_target_domains[i]);
101  neighbor_ranks.push_back(i);
102  }
103  }
104  global_target_domains.clear();
105 
106  // Find the procs to which the sources will be sent.
107  Teuchos::ArrayView<const double> source_point;
108  for ( unsigned source_id = 0;
109  source_id < source_centers.size() / DIM;
110  ++source_id )
111  {
112  source_point = source_centers.view( DIM*source_id, DIM );
113  for ( unsigned b = 0; b < neighbor_target_domains.size(); ++b )
114  {
115  if ( neighbor_target_domains[b].pointInDomain(source_point) )
116  {
117  export_procs.push_back( neighbor_ranks[b] );
118  d_export_ids.push_back( source_id );
119  }
120  }
121  }
122  }
123  DTK_CHECK( d_export_ids.size() == export_procs.size() );
124  d_num_exports = d_export_ids.size();
125 
126  // Create the communication plan.
127  Teuchos::ArrayView<int> export_procs_view = export_procs();
128  d_num_imports = d_distributor->createFromSends( export_procs_view );
129  export_procs.clear();
130 
131 
132  // Unroll the coordinates to handle cases where single source centers
133  // may have multiple destinations.
134  Teuchos::Array<unsigned>::const_iterator export_id_it;
135  Teuchos::Array<double> src_coords( d_num_exports * DIM );
136  for ( int n = 0; n < d_num_exports; ++n )
137  {
138  src_coords( DIM*n, DIM ).assign(
139  source_centers(DIM*d_export_ids[n],DIM) );
140  }
141 
142  // Move the source center coordinates to the target decomposition.
143  Teuchos::ArrayView<const double> src_coords_view = src_coords();
144  target_decomp_source_centers.resize( d_num_imports * DIM );
145  d_distributor->doPostsAndWaits(
146  src_coords_view, DIM, target_decomp_source_centers() );
147 }

154 template<int DIM>
155 template<class T>
156 void CenterDistributor<DIM>::distribute(
157  const Teuchos::ArrayView<const T>& source_decomp_data,
158  const Teuchos::ArrayView<T>& target_decomp_data ) const
159 {
160  DTK_REQUIRE( d_num_imports == target_decomp_data.size() );
161 
162  // Unroll the source data to handle cases where single data points may
163  // have multiple destinations.
164  Teuchos::Array<unsigned>::const_iterator export_id_it;
165  Teuchos::Array<T> src_data( d_num_exports );
166  typename Teuchos::Array<T>::iterator src_it;
167  for ( export_id_it = d_export_ids.begin(),
168  src_it = src_data.begin();
169  export_id_it != d_export_ids.end();
170  ++export_id_it, ++src_it )
171  {
172  DTK_CHECK( *export_id_it < source_decomp_data.size() );
173  *src_it = source_decomp_data[ *export_id_it ];
174  }
175 
176  // Distribute.
177  Teuchos::ArrayView<const T> src_view = src_data();
178  d_distributor->doPostsAndWaits( src_view, 1, target_decomp_data );
179 }

185 template<int DIM>
186 CloudDomain<DIM> CenterDistributor<DIM>::localCloudDomain(
187  const Teuchos::ArrayView<const double>& centers ) const
188 {
189  Teuchos::Array<double> bounds( 2*DIM, 0.0 );
190 
191  if ( centers.size() > 0 )
192  {
193  for ( int d = 0; d < DIM; ++d )
194  {
195  bounds[2*d] = std::numeric_limits<double>::max();
196  bounds[2*d+1] = std::numeric_limits<double>::min();
197  }
198  }
199 
200  Teuchos::ArrayView<const double>::const_iterator center_it;
201  for ( center_it = centers.begin(); center_it != centers.end(); )
202  {
203  for ( int d = 0; d < DIM; ++d )
204  {
205  bounds[2*d] = std::min( bounds[2*d], *(center_it+d) );
206  bounds[2*d+1] = std::max( bounds[2*d+1], *(center_it+d) );
207  }
208  center_it += DIM;
209  }
210 
211  return CloudDomain<DIM>( bounds.getRawPtr() );
212 }
