62 // Given a topological dimension, get the STK entity rank.
63 stk::mesh::EntityRank STKMeshHelpers::getRankFromTopologicalDimension(
64  const int topo_dim, const int space_dim )
65 {
66  stk::mesh::EntityRank stk_rank = stk::topology::INVALID_RANK;
67 
68  switch( space_dim )
69  {
70  case 3:
71  switch( topo_dim )
72  {
73  case 0:
74  stk_rank = stk::topology::NODE_RANK;
75  break;
76  case 1:
77  stk_rank = stk::topology::EDGE_RANK;
78  break;
79  case 2:
80  stk_rank = stk::topology::FACE_RANK;
81  break;
82  case 3:
83  stk_rank = stk::topology::ELEM_RANK;
84  break;
85  default:
86  DTK_CHECK( 0 == topo_dim ||
87  1 == topo_dim ||
88  2 == topo_dim ||
89  3 == topo_dim );
90  break;
91  }
92  break;
93 
94  case 2:
95  switch( topo_dim )
96  {
97  case 0:
98  stk_rank = stk::topology::NODE_RANK;
99  break;
100  case 1:
101  stk_rank = stk::topology::EDGE_RANK;
102  break;
103  case 2:
104  stk_rank = stk::topology::ELEM_RANK;
105  break;
106  default:
107  DTK_CHECK( 0 == topo_dim ||
108  1 == topo_dim ||
109  2 == topo_dim );
110  break;
111  }
112  break;
113 
114  default:
115  stk_rank = stk::topology::NODE_RANK;
116  }
117 
118  return stk_rank;
119 }

122 // Given a STK entity stk_rank, get the topological dimension.
123 int STKMeshHelpers::getTopologicalDimensionFromRank(
124  const stk::mesh::EntityRank stk_rank, const int space_dim )
125 {
126  int topo_dim = 0;
127 
128  switch( space_dim )
129  {
130  case 3:
131  switch( stk_rank )
132  {
133  case stk::topology::NODE_RANK:
134  topo_dim = 0;
135  break;
136  case stk::topology::EDGE_RANK:
137  topo_dim = 1;
138  break;
139  case stk::topology::FACE_RANK:
140  topo_dim = 2;
141  break;
142  case stk::topology::ELEM_RANK:
143  topo_dim = 3;
144  break;
145  default:
146  DTK_CHECK( stk::topology::NODE_RANK == stk_rank ||
147  stk::topology::EDGE_RANK == stk_rank ||
148  stk::topology::FACE_RANK == stk_rank ||
149  stk::topology::ELEM_RANK == stk_rank );
150  break;
151  }
152  break;
153 
154  case 2:
155  switch( stk_rank )
156  {
157  case stk::topology::NODE_RANK:
158  topo_dim = 0;
159  break;
160  case stk::topology::EDGE_RANK:
161  topo_dim = 1;
162  break;
163  case stk::topology::ELEM_RANK:
164  topo_dim = 2;
165  break;
166  default:
167  DTK_CHECK( stk::topology::NODE_RANK == stk_rank ||
168  stk::topology::EDGE_RANK == stk_rank ||
169  stk::topology::ELEM_RANK == stk_rank );
170  break;
171  }
172  break;
173 
174  default:
175  DTK_CHECK( 3 == space_dim || 2 == space_dim );
176  break;
177  }
178 
179  return topo_dim;
180 }
