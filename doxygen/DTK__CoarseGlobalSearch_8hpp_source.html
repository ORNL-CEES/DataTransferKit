133 // Check if two bounding boxes have an intersection.
134 bool CoarseGlobalSearch::boxesIntersect(
135  const Teuchos::Tuple<double,6>& box_A,
136  const Teuchos::Tuple<double,6>& box_B,
137  const double tolerance ) const
138 {
139  double x_tol_A = (box_A[3] - box_A[0])*tolerance;
140  double y_tol_A = (box_A[4] - box_A[1])*tolerance;
141  double z_tol_A = (box_A[5] - box_A[2])*tolerance;
142 
143  double x_tol_B = (box_B[3] - box_B[0])*tolerance;
144  double y_tol_B = (box_B[4] - box_B[1])*tolerance;
145  double z_tol_B = (box_B[5] - box_B[2])*tolerance;
146 
147  return !( ( (box_A[0] - x_tol_A) > (box_B[3] + x_tol_B) ||
148  (box_A[3] + x_tol_A) < (box_B[0] - x_tol_B) ) ||
149  ( (box_A[1] - y_tol_A) > (box_B[4] + y_tol_B) ||
150  (box_A[4] + y_tol_A) < (box_B[1] - y_tol_B) ) ||
151  ( (box_A[2] - z_tol_A) > (box_B[5] + z_tol_B) ||
152  (box_A[5] + z_tol_A) < (box_B[2] - z_tol_B) ) );
153 }

156 // Determine if a point is in a bounding box.
157 bool CoarseGlobalSearch::pointInBox(
158  const Teuchos::ArrayView<const double>& point,
159  const Teuchos::Tuple<double,6>& box,
160  const double tolerance ) const
161 {
162  double x_tol = (box[3] - box[0])*tolerance;
163  double y_tol = (box[4] - box[1])*tolerance;
164  double z_tol = (box[5] - box[2])*tolerance;
165 
166  if( 3 == point.size() )
167  {
168  if ( point[0] >= (box[0] - x_tol) &&
169  point[1] >= (box[1] - y_tol) &&
170  point[2] >= (box[2] - z_tol) &&
171  point[0] <= (box[3] + x_tol) &&
172  point[1] <= (box[4] + y_tol) &&
173  point[2] <= (box[5] + z_tol) )
174  {
175  return true;
176  }
177  }
178  else if( 2 == point.size() )
179  {
180  if ( point[0] >= (box[0] - x_tol) &&
181  point[1] >= (box[1] - y_tol) &&
182  point[0] <= (box[3] + x_tol) &&
183  point[1] <= (box[4] + y_tol) )
184  {
185  return true;
186  }
187  }
188  else if( 1 == point.size() )
189  {
190  if ( point[0] >= (box[0] - x_tol) &&
191  point[0] <= (box[3] + x_tol) )
192  {
193  return true;
194  }
195  }
196 
197  return false;
198 }
