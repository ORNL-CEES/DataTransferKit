68 void ProjectionPrimitives::referenceCellCenter(
69  const shards::CellTopology& cell_topo,
70  Intrepid::FieldContainer<double>& cell_center )
71 {
72  DTK_REQUIRE( 2 == cell_center.rank() );
73  DTK_REQUIRE( Teuchos::as<unsigned>(cell_center.dimension(1)) ==
74  cell_topo.getDimension() );
75 
76  int num_cells = cell_center.dimension(0);
77 
78  switch( cell_topo.getKey() ){
79  case shards::Line<2>::key:
80  case shards::Line<3>::key:
81  for ( int n = 0; n < num_cells; ++n )
82  {
83  cell_center(n,0) = Teuchos::ScalarTraits<double>::zero();
84  }
85  break;
86 
87  case shards::Triangle<3>::key:
88  case shards::Triangle<4>::key:
89  case shards::Triangle<6>::key:
90  for ( int n = 0; n < num_cells; ++n )
91  {
92 
93  cell_center(n,0) = 1.0/3.0;
94  cell_center(n,1) = 1.0/3.0;
95  }
96  break;
97 
98  case shards::Quadrilateral<4>::key:
99  case shards::Quadrilateral<8>::key:
100  case shards::Quadrilateral<9>::key:
101  for ( int n = 0; n < num_cells; ++n )
102  {
103  cell_center(n,0) = Teuchos::ScalarTraits<double>::zero();
104  cell_center(n,1) = Teuchos::ScalarTraits<double>::zero();
105  }
106  break;
107 
108  case shards::Tetrahedron<4>::key:
109  case shards::Tetrahedron<10>::key:
110  case shards::Tetrahedron<11>::key:
111  for ( int n = 0; n < num_cells; ++n )
112  {
113  cell_center(n,0) = 1.0/6.0;
114  cell_center(n,1) = 1.0/6.0;
115  cell_center(n,2) = 1.0/6.0;
116  }
117  break;
118 
119  case shards::Hexahedron<8>::key:
120  case shards::Hexahedron<20>::key:
121  case shards::Hexahedron<27>::key:
122  for ( int n = 0; n < num_cells; ++n )
123  {
124  cell_center(n,0) = Teuchos::ScalarTraits<double>::zero();
125  cell_center(n,1) = Teuchos::ScalarTraits<double>::zero();
126  cell_center(n,2) = Teuchos::ScalarTraits<double>::zero();
127  }
128  break;
129 
130  case shards::Wedge<6>::key:
131  case shards::Wedge<15>::key:
132  case shards::Wedge<18>::key:
133  for ( int n = 0; n < num_cells; ++n )
134  {
135  cell_center(n,0) = 1.0/3.0;
136  cell_center(n,1) = 1.0/3.0;
137  cell_center(n,2) = Teuchos::ScalarTraits<double>::zero();
138  }
139  break;
140 
141  case shards::Pyramid<5>::key:
142  case shards::Pyramid<13>::key:
143  case shards::Pyramid<14>::key:
144  for ( int n = 0; n < num_cells; ++n )
145  {
146  cell_center(n,0) = Teuchos::ScalarTraits<double>::zero();
147  cell_center(n,1) = Teuchos::ScalarTraits<double>::zero();
148  cell_center(n,2) = 1.0/4.0;
149  }
150  break;
151 
152  default:
153  bool cell_topo_supported = false;
154  DTK_INSIST( cell_topo_supported );
155  break;
156  }
157 }

175 bool ProjectionPrimitives::pointInFaceVolumeOfInfluence(
176  const Teuchos::ParameterList& parameters,
177  const Intrepid::FieldContainer<double>& point,
178  const Intrepid::FieldContainer<double>& face_nodes,
179  const Intrepid::FieldContainer<double>& face_node_normals,
180  const shards::CellTopology& face_topology )
181 {
182  DTK_REQUIRE( 1 == point.rank() );
183  DTK_REQUIRE( 2 == face_nodes.rank() );
184  DTK_REQUIRE( 2 == face_node_normals.rank() );
185  DTK_REQUIRE( point.dimension(0) == face_nodes.dimension(1) );
186  DTK_REQUIRE( point.dimension(0) == face_node_normals.dimension(1) );
187  DTK_REQUIRE( face_nodes.dimension(0) == face_node_normals.dimension(0) );
188  DTK_REQUIRE( Teuchos::as<unsigned>(face_nodes.dimension(0)) ==
189  face_topology.getNodeCount() );
190 
191  // Get the spatial dimension
192  int space_dim = point.dimension(0);
193  DTK_CHECK( 3 == space_dim );
194 
195  // Get the geometric tolerance.
196  double geometric_tolerance = parameters.get<double>("Geometric Tolerance");
197 
198  // Project the point onto each bilinear surface formed by the face vertex
199  // normals and the face edges. If the point is on the correct face of each
200  // surface then it is within the volume of influence of the
201  // face. Counter-clockwise ordering of the nodes on the face about the
202  // outward facing normal is required.
203  Intrepid::FieldContainer<double> face_edge_nodes( 2, space_dim );
204  Intrepid::FieldContainer<double> face_edge_node_normals( 2, space_dim );
205  double distance_to_surface = 0.0;
206  int face_num_edges = face_nodes.dimension(0);
207 
208  // First edges.
209  for ( int e = 0; e < face_num_edges-1; ++e )
210  {
211  for ( int i = 0; i < space_dim; ++i )
212  {
213  face_edge_nodes(0,i) = face_nodes(e,i);
214  face_edge_nodes(1,i) = face_nodes(e+1,i);
215  face_edge_node_normals(0,i) = face_node_normals(e,i);
216  face_edge_node_normals(1,i) = face_node_normals(e+1,i);
217  }
218  distance_to_surface = distanceToFaceBilinearSurface(
219  parameters, point, face_edge_nodes, face_edge_node_normals );
220  if ( distance_to_surface > geometric_tolerance )
221  {
222  return false;
223  }
224  }
225 
226  // Last edge.
227  for ( int i = 0; i < space_dim; ++i )
228  {
229  face_edge_nodes(0,i) = face_nodes(face_num_edges-1,i);
230  face_edge_nodes(1,i) = face_nodes(0,i);
231  face_edge_node_normals(0,i) = face_node_normals(face_num_edges-1,i);
232  face_edge_node_normals(1,i) = face_node_normals(0,i);
233  }
234  distance_to_surface = distanceToFaceBilinearSurface(
235  parameters, point, face_edge_nodes, face_edge_node_normals );
236  if ( distance_to_surface > geometric_tolerance )
237  {
238  return false;
239  }
240 
241  // If we got here then the point is in the volume of influence of the
242  // face as it was to the left of all bilinear surfaces formed by the
243  // edges and their node normals.
244  return true;
245 }

275 void ProjectionPrimitives::projectPointToFace(
276  const Teuchos::ParameterList& parameters,
277  const Intrepid::FieldContainer<double>& point,
278  const Intrepid::FieldContainer<double>& face_nodes,
279  const Intrepid::FieldContainer<double>& face_node_normals,
280  const shards::CellTopology& face_topology,
281  Intrepid::FieldContainer<double>& parametric_point,
282  Intrepid::FieldContainer<double>& physical_point,
283  int& face_edge_id,
284  int& face_node_id )
285 {
286  DTK_REQUIRE( 1 == point.rank() );
287  DTK_REQUIRE( 2 == face_nodes.rank() );
288  DTK_REQUIRE( 2 == face_node_normals.rank() );
289  DTK_REQUIRE( 1 == physical_point.rank() );
290  DTK_REQUIRE( point.dimension(0) == face_nodes.dimension(1) );
291  DTK_REQUIRE( point.dimension(0) == face_node_normals.dimension(1) );
292  DTK_REQUIRE( point.dimension(0) == physical_point.dimension(0) );
293  DTK_REQUIRE( face_nodes.dimension(0) ==
294  face_node_normals.dimension(0) );
295  DTK_REQUIRE( Teuchos::as<unsigned>(face_nodes.dimension(0)) ==
296  face_topology.getNodeCount() );
297  DTK_REQUIRE( 2 == parametric_point.rank() );
298  DTK_REQUIRE( 1 == parametric_point.dimension(0) );
299  DTK_REQUIRE( point.dimension(0) == parametric_point.dimension(1) );
300 
301 
302  // Get dimensions.
303  int space_dim = point.dimension(0);
304  int topo_dim = face_topology.getDimension();
305 
306  // Get the geometric tolerance.
307  double geometric_tolerance = parameters.get<double>("Geometric Tolerance");
308 
309  // Get the newton tolerance.
310  double newton_tolerance = parameters.get<double>("Newton Tolerance");
311 
312  // Get the max newton iterations.
313  double max_newton_iters = parameters.get<int>("Max Newton Iterations");
314 
315  // Get the basis functions for the face cell topology.
316  Teuchos::RCP<Intrepid::Basis<double,Intrepid::FieldContainer<double> > >
317  face_basis = IntrepidBasisFactory::create( face_topology );
318 
319  // Initializeunknown vector.
320  parametric_point.initialize( 0.0 );
321 
322  // Set the initial solution guess to the center of the cell for the
323  // parametric coordinates and 0 for distance.
324  int num_points = 1;
325  Intrepid::FieldContainer<double> face_center( num_points, topo_dim );
326  referenceCellCenter( face_topology, face_center );
327  for ( int n = 0; n < topo_dim; ++n )
328  {
329  parametric_point(0,n) = face_center(0,n);
330  }
331  parametric_point(0,topo_dim) = Teuchos::ScalarTraits<double>::zero();
332 
333  // Build the nonlinear problem data.
334  ProjectPointToFaceNonlinearProblem nonlinear_problem(
335  face_basis, point, face_nodes, face_node_normals );
336 
337  // Solve the nonlinear problem.
338  NewtonSolver<ProjectPointToFaceNonlinearProblem>::solve(
339  parametric_point, nonlinear_problem, newton_tolerance, max_newton_iters );
340 
341  // Apply tolerancing. If the point projected near a face edge or
342  // node within the tolerance, move it to that point.
343  if ( std::abs(parametric_point(0,0)) < geometric_tolerance )
344  {
345  parametric_point(0,0) = 0.0;
346  }
347  else if ( std::abs(1.0-parametric_point(0,0)) < geometric_tolerance )
348  {
349  parametric_point(0,0) = 1.0;
350  }
351  else if ( std::abs(1.0+parametric_point(0,0)) < geometric_tolerance )
352  {
353  parametric_point(0,0) = -1.0;
354  }
355  if ( std::abs(parametric_point(0,1)) < geometric_tolerance )
356  {
357  parametric_point(0,1) = 0.0;
358  }
359  else if ( std::abs(1.0-parametric_point(0,1)) < geometric_tolerance )
360  {
361  parametric_point(0,1) = 1.0;
362  }
363  else if ( std::abs(1.0+parametric_point(0,1)) < geometric_tolerance )
364  {
365  parametric_point(0,1) = -1.0;
366  }
367  // This case is for the diagonal of the triangle reference geometry.
368  if ( std::abs(1.0-parametric_point(0,0)-parametric_point(0,1)) <
369  geometric_tolerance )
370  {
371  parametric_point(0,0) = 1.0 - parametric_point(0,1);
372  }
373 
374  // Compute the physical coordinates from the projection in the reference
375  // space.
376  nonlinear_problem.updateState( parametric_point );
377  for ( int d = 0; d < space_dim; ++d )
378  {
379  physical_point(d) = 0.0;
380  for ( int n = 0; n < nonlinear_problem.d_cardinality; ++n )
381  {
382  physical_point(d) +=
383  nonlinear_problem.d_basis_evals(n,0) * face_nodes(n,d);
384  }
385  }
386 
387  // Determine if the point projected onto any of the face nodes or edges.
388  face_edge_id = -1;
389  face_node_id = -1;
390 
391  // Triangle case.
392  if ( (shards::Triangle<3>::key == face_topology.getKey()) ||
393  (shards::Triangle<4>::key == face_topology.getKey()) ||
394  (shards::Triangle<6>::key == face_topology.getKey()) )
395  {
396  if ( (0.0 == parametric_point(0,0)) && (0.0 == parametric_point(0,1)) )
397  {
398  face_node_id = 0;
399  }
400  else if ( (1.0 == parametric_point(0,0)) && (0.0 == parametric_point(0,1)) )
401  {
402  face_node_id = 1;
403  }
404  else if ( (0.0 == parametric_point(0,0)) && (1.0 == parametric_point(0,1)) )
405  {
406  face_node_id = 2;
407  }
408  else if ( 0.0 == parametric_point(0,1) )
409  {
410  face_edge_id = 0;
411  }
412  else if ( 1.0 == parametric_point(0,0) + parametric_point(0,1) )
413  {
414  face_edge_id = 1;
415  }
416  else if ( 0.0 == parametric_point(0,0) )
417  {
418  face_edge_id = 2;
419  }
420  }
421  // Quadrilateral case.
422  else if ( (shards::Quadrilateral<4>::key == face_topology.getKey()) ||
423  (shards::Quadrilateral<8>::key == face_topology.getKey()) ||
424  (shards::Quadrilateral<9>::key == face_topology.getKey()) )
425  {
426  if ( (-1.0 == parametric_point(0,0)) && (-1.0 == parametric_point(0,1)) )
427  {
428  face_node_id = 0;
429  }
430  else if ( (1.0 == parametric_point(0,0)) && (-1.0 == parametric_point(0,1)) )
431  {
432  face_node_id = 1;
433  }
434  else if ( (1.0 == parametric_point(0,0)) && (1.0 == parametric_point(0,1)) )
435  {
436  face_node_id = 2;
437  }
438  else if ( (-1.0 == parametric_point(0,0)) && (1.0 == parametric_point(0,1)) )
439  {
440  face_node_id = 3;
441  }
442  else if( -1.0 == parametric_point(0,1) )
443  {
444  face_edge_id = 0;
445  }
446  else if( 1.0 == parametric_point(0,0) )
447  {
448  face_edge_id = 1;
449  }
450  else if( 1.0 == parametric_point(0,1) )
451  {
452  face_edge_id = 2;
453  }
454  else if( -1.0 == parametric_point(0,0) )
455  {
456  face_edge_id = 3;
457  }
458  }
459  // Unsupported case.
460  else
461  {
462  DTK_INSIST(
463  (shards::Triangle<3>::key == face_topology.getKey()) ||
464  (shards::Triangle<4>::key == face_topology.getKey()) ||
465  (shards::Triangle<6>::key == face_topology.getKey()) ||
466  (shards::Quadrilateral<4>::key == face_topology.getKey()) ||
467  (shards::Quadrilateral<8>::key == face_topology.getKey()) ||
468  (shards::Quadrilateral<9>::key == face_topology.getKey()) );
469  }
470 }

473 /*
474  * \brief Project a feature point to a feature edge. Return false if
475  * the point did not project onto the edge.
476  *
477  * \param parameters Projection parameters.
478  *
479  * \param point Point coordinates (Dim)
480  *
481  * \param edge_nodes Edge node coordinates (Node,Dim)
482  *
483  * \param edge_node_normals Edge node normal vectors (Node,Dim)
484  *
485  * \param projected_point Projected point in physical coordinates (Dim)
486  *
487  * \param edge_node_id The local id of the edge node onto which the
488  * feature node projected. Will be -1 if the feature node did not
489  * project onto any of the edge nodes.
490  *
491  * \return Return true of the feature node projected onto the feature edge.
492  */
493 bool ProjectionPrimitives::projectPointFeatureToEdgeFeature(
494  const Teuchos::ParameterList& parameters,
495  const Intrepid::FieldContainer<double>& point,
496  const Intrepid::FieldContainer<double>& point_normal,
497  const Intrepid::FieldContainer<double>& edge_nodes,
498  const Intrepid::FieldContainer<double>& edge_node_normals,
499  Intrepid::FieldContainer<double>& projected_point,
500  int& edge_node_id )
501 {
502  DTK_REQUIRE( 1 == point.rank() );
503  DTK_REQUIRE( 1 == point_normal.rank() );
504  DTK_REQUIRE( 2 == edge_nodes.rank() );
505  DTK_REQUIRE( 2 == edge_node_normals.rank() );
506  DTK_REQUIRE( 1 == projected_point.rank() );
507  DTK_REQUIRE( point.dimension(0) == edge_nodes.dimension(1) );
508  DTK_REQUIRE( point.dimension(0) == edge_node_normals.dimension(1) );
509  DTK_REQUIRE( point.dimension(0) == projected_point.dimension(0) );
510  DTK_REQUIRE( edge_nodes.dimension(0) ==
511  edge_node_normals.dimension(0) );
512  DTK_REQUIRE( edge_nodes.dimension(0) == 2 );
513 
514  // Get dimensions.
515  int space_dim = point.dimension(0);
516 
517  // Get the geometric tolerance.
518  double geometric_tolerance = parameters.get<double>("Geometric Tolerance");
519 
520  // Get the normal tolerance.
521  double normal_tolerance = parameters.get<double>("Normal Tolerance");
522 
523  // Get the newton tolerance.
524  double newton_tolerance = parameters.get<double>("Newton Tolerance");
525 
526  // Get the max newton iterations.
527  double max_newton_iters = parameters.get<int>("Max Newton Iterations");
528 
529  // Check to make sure that the projection direction is opposite the node
530  // normal directions.
531  double dot_1 = 0.0;
532  double dot_2 = 0.0;
533  for ( int d = 0; d < space_dim; ++d )
534  {
535  dot_1 += point_normal(d)*edge_node_normals(0,d);
536  dot_2 += point_normal(d)*edge_node_normals(1,d);
537  }
538  if ( dot_1 > -0.5*normal_tolerance || dot_2 > -0.5*normal_tolerance )
539  {
540  return false;
541  }
542 
543  // Build the binormals.
544  Intrepid::FieldContainer<double> gvec( space_dim );
545  for ( int d = 0; d < space_dim; ++d )
546  {
547  gvec(d) = edge_nodes(1,d) - edge_nodes(0,d);
548  }
549  Intrepid::FieldContainer<double> normal( space_dim );
550  Intrepid::FieldContainer<double> l( space_dim );
551  Intrepid::FieldContainer<double> edge_node_binormals(
552  edge_node_normals.dimension(0),
553  edge_node_normals.dimension(1) );
554  for ( int n = 0; n < 2; ++n )
555  {
556  for ( int d = 0; d < space_dim; ++d )
557  {
558  normal(d) = edge_node_normals(n,d);
559  }
560  Intrepid::RealSpaceTools<double>::vecprod( l, normal, gvec );
561  for ( int d = 0; d < space_dim; ++d )
562  {
563  edge_node_binormals(n,d) = l(d);
564  }
565  }
566 
567  // Allocate unknown vector.
568  int num_points = 1;
569  Intrepid::FieldContainer<double> u( num_points, space_dim );
570 
571  // Set the initial solution guess to 0.5.
572  for ( int n = 0; n < space_dim; ++n )
573  {
574  u(0,n) = 0.5;
575  }
576 
577  // Build the nonlinear problem data.
578  ProjectPointFeatureToEdgeFeatureNonlinearProblem
579  nonlinear_problem( point, edge_nodes,
580  edge_node_normals, edge_node_binormals );
581 
582  // Solve the nonlinear problem.
583  NewtonSolver<ProjectPointFeatureToEdgeFeatureNonlinearProblem>::solve(
584  u, nonlinear_problem, newton_tolerance, max_newton_iters );
585 
586  // Apply tolerancing.
587  if ( std::abs(u(0,0)) < geometric_tolerance )
588  {
589  u(0,0) = 0.0;
590  }
591  else if ( std::abs(1.0-u(0,0)) < geometric_tolerance )
592  {
593  u(0,0) = 1.0;
594  }
595 
596  // See if the solution was outside the parameteric bounds of the edge.
597  if ( 0.0 > u(0,0) || 1.0 < u(0,0) )
598  {
599  return false;
600  }
601 
602  // Compute the physical coordinates from the projection in the reference
603  // space.
604  for ( int d = 0; d < space_dim; ++d )
605  {
606  projected_point(d) = edge_nodes(0,d) + u(0,0)*gvec(d);
607  }
608 
609  // Check to see if the point projected onto one of the vertices of the
610  // edge.
611  edge_node_id = -1;
612  if ( u(0,0) == 0.0 )
613  {
614  edge_node_id = 0;
615  }
616  else if ( u(0,0) == 1.0 )
617  {
618  edge_node_id = 1;
619  }
620 
621  // There was an interesection so return true.
622  return true;
623 }

654 bool ProjectionPrimitives::edgeEdgeIntersection(
655  const Teuchos::ParameterList& parameters,
656  const Intrepid::FieldContainer<double>& edge_1,
657  const Intrepid::FieldContainer<double>& edge_2,
658  const Intrepid::FieldContainer<double>& edge_2_node_normals,
659  Intrepid::FieldContainer<double>& edge_1_intersection,
660  Intrepid::FieldContainer<double>& edge_2_intersection,
661  int& edge_1_node_id,
662  int& edge_2_node_id )
663 {
664  DTK_REQUIRE( 2 == edge_1.rank() );
665  DTK_REQUIRE( 2 == edge_2.rank() );
666  DTK_REQUIRE( 2 == edge_2_node_normals.rank() );
667  DTK_REQUIRE( 1 == edge_1_intersection.rank() );
668  DTK_REQUIRE( 1 == edge_2_intersection.rank() );

670  DTK_REQUIRE( 2 == edge_2.dimension(0) );
671  DTK_REQUIRE( 2 == edge_2_node_normals.dimension(0) );
672  DTK_REQUIRE( edge_1.dimension(1) == edge_2.dimension(1) );
673  DTK_REQUIRE( edge_1.dimension(1) == edge_2_node_normals.dimension(1) );
674  DTK_REQUIRE( edge_1.dimension(1) == edge_1_intersection.dimension(0) );
675  DTK_REQUIRE( edge_1.dimension(1) == edge_2_intersection.dimension(0) );
676 
677  // The condition number tolerance.
678  double cond_tol = 1.0 / std::sqrt( std::numeric_limits<double>::epsilon() );
679 
680  // Get the spatial dimension.
681  int space_dim = edge_1.dimension(1);
682 
683  // Get the geometric tolerance.
684  double geometric_tolerance = parameters.get<double>("Geometric Tolerance");
685 
686  // Get the merge tolerance.
687  double merge_tolerance = parameters.get<double>("Merge Tolerance");
688 
689  // Build intermediate vectors.
690  Intrepid::FieldContainer<double> bvec( space_dim );
691  Intrepid::FieldContainer<double> gvec( space_dim );
692  Intrepid::FieldContainer<double> dvec( space_dim );
693  Intrepid::FieldContainer<double> d0( space_dim );
694  Intrepid::FieldContainer<double> g0minusb0( space_dim );
695  for ( int i = 0; i < space_dim; ++i )
696  {
697  bvec(i) = edge_1(1,i) - edge_1(0,i);
698  gvec(i) = edge_2(1,i) - edge_2(0,i);
699  dvec(i) = edge_2_node_normals(1,i) - edge_2_node_normals(0,i);
700  d0(i) = edge_2_node_normals(0,i);
701  g0minusb0(i) = edge_2(0,i) - edge_1(0,i);
702  }
703 
704  // Intersection parametric coordinates.
705  double alpha = 0.0;
706  double beta = 0.0;
707 
708  // Compute the parameterized realization of the intersection on edge
709  // 2. Start by building the quadratic equation coefficients.
710  Intrepid::FieldContainer<double> work_vec( space_dim );
711  Intrepid::RealSpaceTools<double>::vecprod( work_vec, bvec, dvec );
712  double a = Intrepid::RealSpaceTools<double>::dot( work_vec, gvec );
713  double b = Intrepid::RealSpaceTools<double>::dot( work_vec, g0minusb0 );
714  Intrepid::RealSpaceTools<double>::vecprod( work_vec, bvec, d0 );
715  b += Intrepid::RealSpaceTools<double>::dot( work_vec, gvec );
716  double c = Intrepid::RealSpaceTools<double>::dot( work_vec, g0minusb0 );
717 
718  // Solve the quadratic equation if a != 0 to get the parameterized
719  // realization of the intersection on the edge 2.
720  if ( std::abs(a) > geometric_tolerance )
721  {
722  double ax2 = 2*a;
723  double sqrt_arg = b*b - 4*a*c;
724 
725  if ( std::abs(sqrt_arg) < geometric_tolerance )
726  {
727  sqrt_arg = 0.0;
728  }
729 
730  // If we get a negative argument for the square root then beta
731  // will be in non-physical (complex) coordinates.
732  if ( sqrt_arg < 0.0 )
733  {
734  return false;
735  }
736 
737  // Compute beta.
738  double sqrt_val = std::sqrt(sqrt_arg);
739  double beta_1 = (-b + sqrt_val) / (ax2);
740  double beta_2 = (-b - sqrt_val) / (ax2);
741 
742  // Resolve inconsistencies by perturbing small beta onto vertices.
743  // These small values will result from the inexact primitive
744  // solutions.
745  if ( std::abs(beta_1) < merge_tolerance )
746  {
747  beta_1 = 0.0;
748  }
749  else if ( std::abs(1.0 - beta_1) < merge_tolerance )
750  {
751  beta_1 = 1.0;
752  }
753  if ( std::abs(beta_2) < merge_tolerance )
754  {
755  beta_2 = 0.0;
756  }
757  else if ( std::abs(1.0 - beta_2) < merge_tolerance )
758  {
759  beta_2 = 1.0;
760  }
761 
762  // If there are two valid solutions then return no intersection as
763  // these edges are nearly parallel.
764  if ( (beta_1 >= 0.0 && beta_1 <= 1.0) &&
765  (beta_2 >= 0.0 && beta_2 <= 1.0) )
766  {
767  return false;
768  }
769 
770  // Choose the solution that is in the accepted range.
771  if ( beta_1 >= 0.0 && beta_1 <= 1.0 )
772  {
773  beta = beta_1;
774  }
775  else if ( beta_2 >= 0.0 && beta_2 <= 1.0 )
776  {
777  beta = beta_2;
778  }
779  else
780  {
781  // There is no intersection if neither solution to the quadratic
782  // equation is between 0 and 1.
783  return false;
784  }
785  }
786  // If a = 0 and b != 0 then the equation is linear.
787  else if ( std::abs(b) > geometric_tolerance )
788  {
789  beta = -c / b;
790 
791  // Resolve inconsistencies by perturbing small beta onto vertices.
792  // These small values will result from the inexact primitive
793  // solutions.
794  if ( std::abs(beta) < merge_tolerance )
795  {
796  beta = 0.0;
797  }
798  else if ( std::abs(1.0 - beta) < merge_tolerance )
799  {
800  beta = 1.0;
801  }
802  }
803  // If a = 0 and b = 0 then beta is undefined.
804  else
805  {
806  return false;
807  }
808 
809  // Check the condition number of beta.
810  Intrepid::FieldContainer<double> nvec( space_dim );
811  double n_length = 0.0;
812  double g_length = 0.0;
813  for ( int i = 0; i < space_dim; ++i )
814  {
815  nvec(i) = g0minusb0(i) + beta*gvec(i);
816  n_length += nvec(i)*nvec(i);
817  g_length += gvec(i)*gvec(i);
818  }
819  Intrepid::FieldContainer<double> ncrossg( space_dim );
820  Intrepid::RealSpaceTools<double>::vecprod( ncrossg, nvec, gvec );
821  double ncrossg_length = 0.0;
822  for ( int i = 0; i < space_dim; ++i )
823  {
824  ncrossg_length += ncrossg(i)*ncrossg(i);
825  }
826 
827  // Only check the condition number if it is defined.
828  if ( ncrossg_length > geometric_tolerance &&
829  n_length > geometric_tolerance )
830  {
831  double beta_cond = std::sqrt(n_length) * std::sqrt(g_length) /
832  std::sqrt(ncrossg_length);
833  if ( beta_cond > cond_tol )
834  {
835  return false;
836  }
837  }
838 
839  // Compute the parameterized realization of the intersection on the edge
840  // 1.
841  Intrepid::FieldContainer<double> l( space_dim );
842  Intrepid::RealSpaceTools<double>::scale( dvec, beta );
843  Intrepid::RealSpaceTools<double>::add( work_vec, d0, dvec );
844  Intrepid::RealSpaceTools<double>::vecprod( l, gvec, work_vec );
845  double dot1 = Intrepid::RealSpaceTools<double>::dot( l, g0minusb0 );
846  double dot2 = Intrepid::RealSpaceTools<double>::dot( l, bvec );
847 
848  if ( std::abs(dot2) < geometric_tolerance )
849  {
850  return false;
851  }
852  alpha = dot1 / dot2;
853 
854  // Resolve inconsistencies by pertubing small alpha onto vertices. These
855  // small values will result from the inexact primitive solutions.
856  if ( std::abs(alpha) < merge_tolerance )
857  {
858  alpha = 0.0;
859  }
860  else if ( std::abs(1.0 - alpha) < merge_tolerance )
861  {
862  alpha = 1.0;
863  }
864 
865  // Check the condition number of alpha.
866  Intrepid::FieldContainer<double> lvec( space_dim );
867  double l_length = 0.0;
868  double b_length = 0.0;
869  for ( int i = 0; i < space_dim; ++i )
870  {
871  lvec(i) = alpha*bvec(i) - g0minusb0(i);
872  l_length += lvec(i)*lvec(i);
873  b_length += bvec(i)*bvec(i);
874  }
875  Intrepid::FieldContainer<double> lcrossb( space_dim );
876  Intrepid::RealSpaceTools<double>::vecprod( lcrossb, lvec, bvec );
877  double lcrossb_length = 0.0;
878  for ( int i = 0; i < space_dim; ++i )
879  {
880  lcrossb_length += lcrossb(i)*lcrossb(i);
881  }
882 
883  // Only check the condition number if it is defined.
884  if ( lcrossb_length > geometric_tolerance && l_length > geometric_tolerance )
885  {
886  double alpha_cond = std::sqrt(l_length) * std::sqrt(b_length) /
887  std::sqrt(lcrossb_length);
888  if ( alpha_cond > cond_tol )
889  {
890  return false;
891  }
892  }
893 
894  // Check for an intersection on the edge 1.
895  if ( alpha < 0.0 || alpha > 1.0 )
896  {
897  return false;
898  }
899 
900  // Check for an intersection on the edge 2.
901  if ( beta < 0.0 || beta > 1.0 )
902  {
903  return false;
904  }
905 
906  // Build the physical intersection location on the edge 2.
907  for ( int i = 0; i < space_dim; ++i )
908  {
909  edge_2_intersection(i) = edge_2(0,i) + beta*gvec(i);
910  }
911 
912  // Build the physical intersection location on the edge 1.
913  for ( int i = 0; i < space_dim; ++i )
914  {
915  edge_1_intersection(i) = edge_1(0,i) + alpha*bvec(i);
916  }
917 
918  // Check the error in the solution.
919  double d_length = 0.0;
920  Intrepid::FieldContainer<double> v_vec( space_dim );
921  for ( int i = 0; i < space_dim; ++i )
922  {
923  d_length += work_vec(i)*work_vec(i);
924  v_vec(i) = edge_1_intersection(i) - edge_2_intersection(i);
925  }
926  d_length *= -d_length;
927  d_length += 1.0;
928  for ( int i = 0; i < space_dim; ++i )
929  {
930  v_vec(i) *= d_length;
931  }
932  double v_length = 0.0;
933  for ( int i = 0; i < space_dim; ++i )
934  {
935  v_length += v_vec(i)*v_vec(i);
936  }
937  double error = std::sqrt(v_length) /
938  std::min( std::sqrt(b_length), std::sqrt(g_length) );
939  if ( error > cond_tol ||
940  b_length < geometric_tolerance ||
941  g_length < geometric_tolerance )
942  {
943  return false;
944  }
945 
946  // Determine if any of the edge 1 nodes were intersected.
947  if ( 0.0 == alpha )
948  {
949  edge_1_node_id = 0;
950  }
951  else if ( 1.0 == alpha )
952  {
953  edge_1_node_id = 1;
954  }
955  else
956  {
957  edge_1_node_id = -1;
958  }
959 
960  // Determine if any of the edge 2 nodes were intersected.
961  if ( 0.0 == beta )
962  {
963  edge_2_node_id = 0;
964  }
965  else if ( 1.0 == beta )
966  {
967  edge_2_node_id = 1;
968  }
969  else
970  {
971  edge_2_node_id = -1;
972  }
973 
974  // Both alpha and beta are between 0 and 1 so there is an
975  // intersection.
976  return true;
977 }

996 ProjectionPrimitives::distanceToFaceBilinearSurface(
997  const Teuchos::ParameterList& parameters,
998  const Intrepid::FieldContainer<double>& point,
999  const Intrepid::FieldContainer<double>& face_edge_nodes,
1000  const Intrepid::FieldContainer<double>& face_edge_node_normals )

1003  double geometric_tolerance = parameters.get<double>("Geometric Tolerance");
1004 
1005  // Get the newton tolerance.
1006  double newton_tolerance = parameters.get<double>("Newton Tolerance");
1007 
1008  // Get the max newton iterations.
1009  double max_newton_iters = parameters.get<int>("Max Newton Iterations");
1010 
1011  // Compute the scale factor. Choose half the length of the face for
1012  // simplicity.
1013  double xdist = face_edge_nodes(1,0) - face_edge_nodes(0,0);
1014  double ydist = face_edge_nodes(1,1) - face_edge_nodes(0,1);
1015  double zdist = face_edge_nodes(1,2) - face_edge_nodes(0,2);
1016  double c = 0.5*std::sqrt( xdist*xdist + ydist*ydist + zdist*zdist );
1017

1019  Intrepid::FieldContainer<double> u( 1, point.dimension(0) );
1020  u(0,0) = 0.5;
1021  u(0,1) = 0.5;
1022  u(0,2) = Teuchos::ScalarTraits<double>::zero();
1023  PointInFaceVolumeOfInfluenceNonlinearProblem nonlinear_problem(
1024  point, face_edge_nodes, face_edge_node_normals, c );
1025  NewtonSolver<PointInFaceVolumeOfInfluenceNonlinearProblem>::solve(
1026  u, nonlinear_problem, newton_tolerance, max_newton_iters );
1027 
1028  // Check for degeneracy.
1029  nonlinear_problem.updateState( u );
1030  Intrepid::FieldContainer<double> v1( nonlinear_problem.d_space_dim );
1031  Intrepid::FieldContainer<double> v2( nonlinear_problem.d_space_dim );
1032  for ( int i = 0; i < nonlinear_problem.d_space_dim; ++i )
1033  {
1034  v1(i) = nonlinear_problem.d_face_edge_nodes(1,i) -
1035  nonlinear_problem.d_face_edge_nodes(0,i);
1036  v2(i) = v1(i) + nonlinear_problem.d_c*u(0,1)*(
1037  nonlinear_problem.d_face_edge_node_normals(1,i) -
1038  nonlinear_problem.d_face_edge_node_normals(0,i) );
1039  }
1040 
1041  // Return a positive number if degenerate.
1042  if ( Intrepid::RealSpaceTools<double>::dot(v1,v2) < geometric_tolerance )
1043  {
1044  return 1.0;
1045  }
1046 
1047  // Return the solution if not degenerate.
1048  return u(0,2);
1049 }
