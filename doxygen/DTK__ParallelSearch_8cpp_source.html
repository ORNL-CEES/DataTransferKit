49 // Constructor.
50 ParallelSearch::ParallelSearch(
51  const Teuchos::RCP<const Teuchos::Comm<int> >& comm,
52  const int physical_dimension,
53  const EntityIterator& domain_iterator,
54  const Teuchos::RCP<EntityLocalMap>& domain_local_map,
55  const Teuchos::ParameterList& parameters )
56  : d_comm( comm )
57  , d_physical_dim( physical_dimension )
58  , d_track_missed_range_entities( false )
59  , d_missed_range_entity_ids( 0 )
60 {
61  // Set the parameters with the local map.
62  domain_local_map->setParameters( parameters );
63 
64  // Determine if we are tracking missed range entities.
65  if ( parameters.isParameter("Track Missed Range Entities") )
66  {
67  d_track_missed_range_entities =
68  parameters.get<bool>("Track Missed Range Entities");
69  }
70 
71  // Build a coarse global search as this object must be collective across
72  // the communicator.
73  d_coarse_global_search = Teuchos::rcp(
74  new CoarseGlobalSearch(d_comm, physical_dimension,
75  domain_iterator, parameters) );
76 
77  // Only do the local search if there are local domain entities.
78  d_empty_domain = ( 0 == domain_iterator.size() );
79  if ( !d_empty_domain )
80  {
81  d_coarse_local_search = Teuchos::rcp(
82  new CoarseLocalSearch(domain_iterator, domain_local_map, parameters) );
83  d_fine_local_search = Teuchos::rcp(
84  new FineLocalSearch(domain_local_map) );
85  }
86 }

89 // Search the domain with the range entity centroids and construct the
90 // graph. This will update the state of the object.
91 void ParallelSearch::search(
92  const EntityIterator& range_iterator,
93  const Teuchos::RCP<EntityLocalMap>& range_local_map,
94  const Teuchos::ParameterList& parameters )
95 {
96  // Set the parameters with the local map.
97  range_local_map->setParameters( parameters );
98 
99  // Empty range flag.
100  d_empty_range = ( 0 == range_iterator.size() );
101 
102  // Reset the state of the object.
103  d_range_owner_ranks.clear();
104  d_domain_to_range_map.clear();
105  d_range_to_domain_map.clear();
106  d_parametric_coords.clear();
107 
108  // Perform a coarse global search to redistribute the range entities.
109  Teuchos::Array<EntityId> range_entity_ids;
110  Teuchos::Array<int> range_owner_ranks;
111  Teuchos::Array<double> range_centroids;
112  d_coarse_global_search->search(
113  range_iterator, range_local_map, parameters,
114  range_entity_ids, range_owner_ranks, range_centroids );
115 
116  // If needed, extract the range entities that were missed during the
117  // coarse global search.
118  Teuchos::Array<EntityId> found_range_entity_ids;
119  Teuchos::Array<int> found_range_ranks;
120  Teuchos::Array<EntityId> missed_range_entity_ids;
121  Teuchos::Array<int> missed_range_ranks;
122  if ( d_track_missed_range_entities )
123  {
124  missed_range_entity_ids = Teuchos::Array<EntityId>(
125  d_coarse_global_search->getMissedRangeEntityIds() );
126  missed_range_ranks.assign( missed_range_entity_ids.size(),
127  d_comm->getRank() );
128  }
129 
130  // Only do the local search if there are local domain entities.
131  Teuchos::Array<int> export_range_ranks;
132  Teuchos::Array<EntityId> export_data;
133  if ( !d_empty_domain )
134  {
135  // For each range centroid, perform a local search.
136  int num_range = range_entity_ids.size();
137  Teuchos::Array<Entity> domain_neighbors;
138  Teuchos::Array<Entity> domain_parents;
139  Teuchos::Array<double> reference_coordinates;
140  Teuchos::Array<double> local_coords( d_physical_dim );
141  int num_parents = 0;
142  for ( int n = 0; n < num_range; ++n )
143  {
144  // Perform a coarse local search to get the nearest domain
145  // entities to the point.
146  d_coarse_local_search->search(
147  range_centroids(d_physical_dim*n,d_physical_dim),
148  parameters,
149  domain_neighbors );
150 
151  // Perform a fine local search to get the entities the point maps
152  // to.
153  d_fine_local_search->search(
154  domain_neighbors,
155  range_centroids(d_physical_dim*n,d_physical_dim),
156  parameters,
157  domain_parents,
158  reference_coordinates );
159 
160  // Store the potentially multiple parametric realizations of the
161  // point.
162  std::unordered_map<EntityId,Teuchos::Array<double> > ref_map;
163  num_parents = domain_parents.size();
164  for ( int p = 0; p < num_parents; ++p )
165  {
166  // Store the range data in the domain parallel decomposition.
167  local_coords().assign(
168  reference_coordinates(d_physical_dim*p,d_physical_dim) );
169  d_range_owner_ranks.emplace(
170  range_entity_ids[n], range_owner_ranks[n] );
171  d_domain_to_range_map.emplace(
172  domain_parents[p].id(), range_entity_ids[n] );
173  ref_map.emplace(
174  domain_parents[p].id(), local_coords );
175 
176  // Extract the data to communicate back to the range parallel
177  // decomposition.
178  export_range_ranks.push_back( range_owner_ranks[n] );
179  export_data.push_back( range_entity_ids[n] );
180  export_data.push_back( domain_parents[p].id() );
181  export_data.push_back(
182  Teuchos::as<EntityId>(d_comm->getRank()) );
183  }
184 
185  // If we found parents for the point, store them.
186  if ( num_parents > 0 )
187  {
188  d_parametric_coords.emplace( range_entity_ids[n], ref_map );
189 
190  // If we are tracking missed entities, also track those that
191  // we found so we can determine if an entity was found after
192  // being sent to multiple destinations.
193  if ( d_track_missed_range_entities )
194  {
195  found_range_entity_ids.push_back( range_entity_ids[n] );
196  found_range_ranks.push_back( range_owner_ranks[n] );
197  }
198  }
199 
200  // Otherwise, if we are tracking missed entities report this.
201  else if ( d_track_missed_range_entities )
202  {
203  missed_range_entity_ids.push_back( range_entity_ids[n] );
204  missed_range_ranks.push_back( range_owner_ranks[n] );
205  }
206  }
207  }
208 
209  // Back-communicate the domain entities in which we found each range
210  // entity to complete the mapping.
211  Tpetra::Distributor domain_to_range_dist( d_comm );
212  int num_import =
213  domain_to_range_dist.createFromSends( export_range_ranks() );
214  Teuchos::Array<EntityId> domain_data( 3*num_import );
215  Teuchos::ArrayView<const EntityId> export_data_view = export_data();
216  domain_to_range_dist.doPostsAndWaits( export_data_view, 3, domain_data() );
217 
218  // Store the domain data in the range parallel decomposition.
219  for ( int i = 0; i < num_import; ++i )
220  {
221  d_domain_owner_ranks.emplace(
222  domain_data[3*i+1], domain_data[3*i+2] );
223  d_range_to_domain_map.emplace(
224  domain_data[3*i], domain_data[3*i+1] );
225  }
226 
227  // If we are tracking missed entities, back-communicate the missing entities
228  // and found entities to determine which entities are actually missing.
229  if ( d_track_missed_range_entities )
230  {
231  // Back-communicate the missing entities.
232  Tpetra::Distributor missed_range_dist( d_comm );
233  int num_import_missed =
234  missed_range_dist.createFromSends( missed_range_ranks() );
235  Teuchos::Array<EntityId> import_missed( num_import_missed );
236  Teuchos::ArrayView<const EntityId> missed_view =
237  missed_range_entity_ids();
238  missed_range_dist.doPostsAndWaits( missed_view, 1, import_missed() );
239 
240  // Back-communicate the found entities.
241  Tpetra::Distributor found_range_dist( d_comm );
242  int num_import_found =
243  found_range_dist.createFromSends( found_range_ranks() );
244  Teuchos::Array<EntityId> import_found( num_import_found );
245  Teuchos::ArrayView<const EntityId> found_view =
246  found_range_entity_ids();
247  found_range_dist.doPostsAndWaits( found_view, 1, import_found() );
248 
249 
250  // Create a unique list of missed entities.
251  std::sort( import_missed.begin(), import_missed.end() );
252  auto import_missed_unique_end = std::unique(
253  import_missed.begin(), import_missed.end());
254  import_missed.resize(std::distance(import_missed.begin(), import_missed_unique_end));
255 
256  // Create a unique list of found entities.
257  std::sort( import_found.begin(), import_found.end() );
258  auto import_found_unique_end = std::unique(
259  import_found.begin(), import_found.end());
260  import_found.resize(std::distance(import_found.begin(), import_found_unique_end));
261 
262  // Intersect the found and missed entities to determine if there are any
263  // that were found on one process but missed on another.
264  Teuchos::Array<EntityId> false_positive_missed( import_missed.size() );
265  auto false_positive_end =
266  std::set_intersection( import_missed.begin(), import_missed.end(),
267  import_found.begin(), import_found.end(),
268  false_positive_missed.begin() );
269 
270  // Create a list of missed entities without the false positives.
271  d_missed_range_entity_ids.resize( num_import_missed );
272  auto missed_range_end = std::set_difference(
273  import_missed.begin(), import_missed.end(),
274  false_positive_missed.begin(), false_positive_end,
275  d_missed_range_entity_ids.begin() );
276  d_missed_range_entity_ids.resize(
277  std::distance(d_missed_range_entity_ids.begin(),
278  missed_range_end) );
279 
280 #if HAVE_DTK_DBC
281  unsigned long long int n_entities = d_missed_range_entity_ids.size() +
282  import_found.size();
283  DTK_REQUIRE( n_entities == range_iterator.size() );
284 #endif
285  }

289 // Given a domain entity id, get the ids of the range entities that mapped to it.
290 void ParallelSearch::getRangeEntitiesFromDomain(
291  const EntityId domain_id,
292  Teuchos::Array<EntityId>& range_ids ) const
293 {
294  DTK_REQUIRE( !d_empty_domain );
295  auto domain_pair = d_domain_to_range_map.equal_range( domain_id );
296  range_ids.resize( std::distance(domain_pair.first,domain_pair.second) );
297  auto range_it = range_ids.begin();
298  for ( auto domain_it = domain_pair.first;
299  domain_it != domain_pair.second;
300  ++domain_it, ++range_it )
301  {
302  *range_it = domain_it->second;
303  }
304 }
