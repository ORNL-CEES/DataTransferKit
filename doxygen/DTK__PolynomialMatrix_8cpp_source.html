72 // Apply operation.
73 void PolynomialMatrix::apply(
74  const Tpetra::MultiVector<double,int,SupportId> &X,
75  Tpetra::MultiVector<double,int,SupportId> &Y,
76  Teuchos::ETransp mode,
77  double alpha,
78  double beta ) const
79 {
80  DTK_REQUIRE( d_domain_map->isSameAs(*(X.getMap())) );
81  DTK_REQUIRE( d_range_map->isSameAs(*(Y.getMap())) );
82  DTK_REQUIRE( X.getNumVectors() == Y.getNumVectors() );
83 
84  // Get the size of the problem and view of the local vectors.
85  int poly_size = d_polynomial->getNumVectors();
86  int num_vec = X.getNumVectors();
87  int local_length = d_polynomial->getLocalLength();
88  Teuchos::ArrayRCP<Teuchos::ArrayRCP<const double> > poly_view =
89  d_polynomial->get2dView();
90  Teuchos::ArrayRCP<Teuchos::ArrayRCP<double> > y_view =
91  Y.get2dViewNonConst();
92 
93  // Scale Y by beta.
94  Y.scale( beta );
95 
96  // No transpose.
97  if ( Teuchos::NO_TRANS == mode )
98  {
99  // Broadcast the polynomial components of X from the root rank.
100  Teuchos::Array<double> x_poly( poly_size * num_vec, 0.0 );
101  if ( 0 == d_comm()->getRank() )
102  {
103  Teuchos::ArrayRCP<Teuchos::ArrayRCP<const double> > x_view =
104  X.get2dView();
105  for ( int n = 0; n < num_vec; ++n )
106  {
107  std::copy( &x_view[n][0], &x_view[n][0] + poly_size,
108  &x_poly[n*poly_size] );
109  }
110  }
111  Teuchos::broadcast( *d_comm, 0, x_poly() );
112 
113  // Do the local mat-vec.
114  int stride = 0;
115  for ( int n = 0; n < num_vec; ++n )
116  {
117  stride = n*poly_size;
118 
119  for ( int i = 0; i < local_length; ++i )
120  {
121  for ( int p = 0; p < poly_size; ++p )
122  {
123  y_view[n][i] += alpha * poly_view[p][i] * x_poly[stride + p];
124  }
125  }
126  }
127  }
128 
129  // Transpose.
130  else if ( Teuchos::TRANS == mode )
131  {
132  // Make a work vector.
133  Tpetra::MultiVector<double,int,SupportId> work( Y.getMap(),
134  Y.getNumVectors() );
135 
136  // Export X to the polynomial decomposition.
137  Tpetra::Export<int,SupportId> exporter( X.getMap(), work.getMap() );
138  work.doExport( X, exporter, Tpetra::INSERT );
139 
140  // Do the local mat-vec.
141  Teuchos::ArrayRCP<Teuchos::ArrayRCP<double> > work_view =
142  work.get2dViewNonConst();
143  Teuchos::Array<double> products( poly_size * num_vec, 0.0 );
144  int stride = 0;
145  for ( int n = 0; n < num_vec; ++n )
146  {
147  stride = n*poly_size;
148  for ( int p = 0; p < poly_size; ++p )
149  {
150  for ( int i = 0; i < local_length; ++i )
151  {
152  products[stride + p] += poly_view[p][i] * work_view[n][i];
153  }
154  }
155  }
156 
157  // Reduce the results back to the root rank.
158  Teuchos::Array<double> product_sums( poly_size * num_vec, 0.0 );
159 #ifdef HAVE_MPI
160  Teuchos::RCP< const Teuchos::MpiComm<int> > mpi_comm =
161  Teuchos::rcp_dynamic_cast< const Teuchos::MpiComm<int> >( d_comm );
162  Teuchos::RCP< const Teuchos::OpaqueWrapper<MPI_Comm> > opaque_comm =
163  mpi_comm->getRawMpiComm();
164  MPI_Comm raw_comm = (*opaque_comm)();
165  MPI_Reduce( products.getRawPtr(),
166  product_sums.getRawPtr(),
167  poly_size * num_vec,
168  MPI_DOUBLE,
169  MPI_SUM,
170  0,
171  raw_comm );
172 #else
173  product_sums = products;
174 #endif
175 
176  // Assign the values to Y on the root rank.
177  if ( 0 == d_comm->getRank() )
178  {
179  for ( int n = 0; n < num_vec; ++n )
180  {
181  for ( int p = 0; p < poly_size; ++p )
182  {
183  y_view[n][p] += alpha * product_sums[n*poly_size+p];
184  }
185  }
186  }
187  }
188 
189  else
190  {
191  DTK_INSIST( mode == Teuchos::NO_TRANS || mode == Teuchos::TRANS );
192  }
193 }
